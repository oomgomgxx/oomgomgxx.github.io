Dubbo SPI机制源码浅析

 2020-08-12   作者: TanDi   类型：原创

  SPI 的全称为 Service Provider Interface（服务提供者接口）。SPI 是模块化编程中重要的基础，我们可以依赖 SPI 机制做到不修改代码的情况下给接口切换实现类，以此来提高应用的可扩展性。

  SPI 并不是 Dubbo 中独有的机制，准确来说它扩展自 JDK SPI。除 Dubbo 之外，其实很多常见的开源框架亦都直接或间接地在使用 SPI 机制。譬如 Hibernate 第三方库加载（缓存、连接池等）， Sping Boot Starter 自动装配（SpringFactoriesLoader），还有 Servlet 3.0 的无 xml 容器配置方式等，都是建基于 SPI 思想之上实现的。而在 JDK 内部对 SPI 的最佳实践莫过于是 JDBC 驱动加载，自 JDBC 4.0 规范（JDK1.6版本）开始就不再需要手动通过 Class.forname 来加载驱动，取而代之的是通过 SPI 机制检索类路径下的关联文件来自动加载驱动实现。

JDK SPI 入门

简介

  在了解 Dubbo SPI 机制之前，个人认为很有必要先入门一下 JDK SPI，因为这是很多 Java 开源框架实现模块化编程的编程思想。

快速入门案例

一、创建 SPI 扩展接口

Person：全限定名为 spi.Person

public interface Person {
    void sayHi();
}

二、给 SPI 扩展接口提供可选实现

Student：全限定名为 spi.Student

public class Student implements Person {
    @Override
    public void sayHi() {
        System.out.println("你好，我是学无止境的小学鸡");
    }
}

Worker：全限定名为 spi.Worker

public class Worker implements Person {
    @Override
    public void sayHi() {
        System.out.println("你好，我是玉树临风的工作人员");
    }
}

三、提供关联文件

  1. 在类路径 WEB-INF/services/ 下，创建一个以 SPI 扩展接口全限定名为名称的普通文件（当前案例为 spi.Person）
  2. 将实现类的全限定名填写进去，当前例子如下：
spi.Worker

注意：关联文件中可以填写多个实现类全限定名，每个类全限定名占一行

ServiceLoader 源码浅析

字段、构造函数

public final class ServiceLoader<S> implements Iterable<S>
{
    // 关联文件存放位置
    private static final String PREFIX = "META-INF/services/";

    // SPI扩展接口Class实例
    private final Class<S> service;

    // 加载关联文件的类加载器
    // 注意：因为同一个类加载器“实例”不会对同一个类文件加载两次
    // 所以在类全先定位一致的情况下，要是T.class被加载过，那么即使修改T.class后再reload也不会加载到新的T.class
    // 而只有T.class被卸载(实例都被回收且没有如何引用)或加载T.class的类加载器被回收后，在再次加载时才会加载到新的T.class
    private final ClassLoader loader;

    // 安全访问控制，和SecurityManager配置相关
    private final AccessControlContext acc;

    // 实现的对象实例缓存器（容器单例）
    private LinkedHashMap<String,S> providers = new LinkedHashMap<>();

    // 懒加载形式的服务发现迭代器
    // ServiceLoader回先从providers查找，找不到才从LazyIterator中发现
    private LazyIterator lookupIterator;

    // 自由构造函数，也就是说实例是在ServiceLoader内部负责创建的
    private ServiceLoader(Class<S> svc, ClassLoader cl) {
        service = Objects.requireNonNull(svc, "Service interface cannot be null");
        loader = (cl == null) ? ClassLoader.getSystemClassLoader() : cl;
        acc = (System.getSecurityManager() != null) ? AccessController.getContext() : null;
        // 没创建一次实例都会reload一次
        // 既清除加载缓存，并重新创建LazyIterator
        reload();
    }
    //...
}

load 方法

public static <S> ServiceLoader<S> load(Class<S> service) {
    // 获取当前操作类对应的类加载器
    ClassLoader cl = Thread.currentThread().getContextClassLoader(); 
    return ServiceLoader.load(service, cl);
}

public static <S> ServiceLoader<S> load(Class<S> service,
                                        ClassLoader loader)
{
    // 创建ServiceLoader实例
    return new ServiceLoader<>(service, loader); 
}

reload 方法

// 废弃缓存，并重新创建懒加载形式的服务发现迭代器
public void reload() {
    // 1. 清理本地缓存（Map）
    providers.clear(); 
    // 2. 重新创建LazyIterator
    // 注意：之所以重新创建LazyIterator，是为了让ServiceLoader的iterator()方法重新回到懒加载的效果。懒加载的核心是：只有在使用时才会去加载，是一种避免不必要消耗的处理手法
    // 参数1：SPI扩展接口类Class实例；参数2：类加载器
    lookupIterator = new LazyIterator(service, loader); 
}

iterator 方法

public Iterator<S> iterator() {
    return new Iterator<S>() {
        // 从缓存(Map)中得到元素迭代器
        Iterator<Map.Entry<String,S>> knownProviders
            = providers.entrySet().iterator();
        // 当缓存(Map)中找不到时，才会用LazyIterator发现
        public boolean hasNext() {
            if (knownProviders.hasNext())
                return true;
            return lookupIterator.hasNext();
        }

        public S next() {
            if (knownProviders.hasNext())
                return knownProviders.next().getValue();
            return lookupIterator.next();
        }

        public void remove() {
            throw new UnsupportedOperationException();
        }

    };
}

LazyIterator 懒加载服务发现

// 迭代器模式
private class LazyIterator
    implements Iterator<S>
{
    // 通过URLClassLoader加载出来的内容
    Enumeration<URL> configs = null;
    // 通过configs解析出来的实现信息
    Iterator<String> pending = null;
    // 关联文件中待加载的实现类全限定名
    String nextName = null;

    // 加载SPI扩展接口实现的主要逻辑
    private boolean hasNextService() {
        if (nextName != null) {
            return true;
        }
        if (configs == null) {
            try { 
                // 拼装管理文件路径
                String fullName = PREFIX + service.getName();
                // 如果没有指定类加载器，则使用应用(系统)类加载器尝试加载
                if (loader == null) 
                    configs = ClassLoader.getSystemResources(fullName);
                // 通过指定类加载器加载
                // 注意：ExtClassLoader、AppClassLoader继承自URLClassLoader，而URLClassLoader用于加载直径路径下jar或文件夹中的Class
                else 
                    configs = loader.getResources(fullName);
            }
            // ...
        }
        while ((pending == null) || !pending.hasNext()) {
            if (!configs.hasMoreElements()) {
                return false; // 表示类加载器并没有加载到内容
            }
            // 将加载内容解析到Iterator中
            pending = parse(service, configs.nextElement());
        }
        nextName = pending.next();
        return true;
    }

    private S nextService() {
        // Fail Fast
        // 如果类加载器没有加载到内容，则直接抛出NoSuchElementException异常
        if (!hasNextService())
            throw new NoSuchElementException();

        String cn = nextName;
        nextName = null;
        Class<?> c = null;
        try {
            // 获取到SPI扩展接口实现的Class对象
            // 注意，可以看到加载并不会初始化类
            c = Class.forName(cn, false, loader); 
        }
        // ...
        try {
            // 将实现的Class对象实例化，并保存到ServiceLoader的缓存providers中
            S p = service.cast(c.newInstance());
            providers.put(cn, p);
            return p;
        } 
        // ...
    }

    public boolean hasNext() {
        // 如果加载的内容没有访问控制，则直接用hasNextService()获取
        if (acc == null) {
            return hasNextService();
        // 反之包装hasNextService()进行访问控制检验
        } else {
            PrivilegedAction<Boolean> action = new PrivilegedAction<Boolean>() {
                public Boolean run() { 
                    return hasNextService(); 
                }
            };
            return AccessController.doPrivileged(action, acc);
        }
    }
    // ...
}

三、提供关联文件

  1. 在类路径 WEB-INF/services/ 下，创建一个以 SPI 扩展接口全限定名为名称的普通文件（当前案例为 spi.Person）

四、使用 Dubbo SPI

普通模式使用：直接依赖 @SPI 操作

  • @SPI 注解用于标注扩展点（SPI扩展接口）
@Test
public void testSPI() {
    ExtensionLoader<Person> loader = ExtensionLoader.getExtensionLoader(Person.class);
    // 方式1：获取默认的扩展实现，既 @SPI("默认实现的key值")
    //Person person = loader.getDefaultExtension();
    // 方式2：先根据key获取，如果找不到就获取默认值
    String key = "batman";
    Person person = loader.getOrDefaultExtension(key);
    person.sayHi();
}

适配模式使用：@SPI + @Adaptive

  • @Adaptive 注解用于启用 Adaptive 适配模式
  • @Adaptive 注解可以标注在扩展点抽象方法上，也可以标注在扩展点实现类上。当注解标注在扩展点实现类上时，表示为这事一个装饰器类（参考：AdaptiveCompiler 类）
  • 被 @Adaptive 注解标注的抽象方法，必需在参数列表中添加 Dubbo 提供的 URL 类作为入参，否则会抛出 IllegalStateException 异常。URL 的主要作用是为面向切面编程提供条件依据

步骤1：开启 Adaptive 适配模式的前提除了添加注解外，还要在扩展点抽象方法上和其实现类方法上添加 @Adaptive 注解，如下：

import org.apache.dubbo.common.URL;
import org.apache.dubbo.common.extension.Adaptive;
import org.apache.dubbo.common.extension.SPI;

@SPI("student")
public interface Person {
    // 启用适配模式
    // 其中value值为URL参数名称，而该参数的值将会作为AOP适配实现类的依据
    @Adaptive(value = "active") 
    void sayHi(URL url);
}

public class Worker implements Person {
    static {
        System.out.println("Worker类被初始化");
    }
    @Override
    public void sayHi(URL url) {
        System.out.println("你好，我是玉树临风的工作人员");
    }
}

public class Student implements Person {
    static {
        System.out.println("Student类被初始化");
    }
    @Override
    public void sayHi(URL url) {
        System.out.println("你好，我是学无止境的小学鸡");
    }
}

步骤2：使用

@Test
public void testSPI() {

    ExtensionLoader<Person> loader = ExtensionLoader.getExtensionLoader(Person.class);
    // 使用适配模式匹配
    // 因为使用AOP思想实现，所以这里返回的实质是一个代理实例
    Person person = loader.getAdaptiveExtension(); 

    String request = "http://www.baidu.com?";
    // 表示激活key为worker的扩展点实现

// Student类被初始化
// 你好，我是玉树临风的工作人员

激活模式使用：@SPI + @Activate

  • 激活模式和适配模式类似，都需要特定的条件下才能启动扩展接口实现
  • @Activate 注解可以标注在扩展点实现类上或其方法上，@Activate 常见用法有两种：
    1. @Activate (value = “”)：指定 value 值，当 getActivateExtension 时 URL 中出现其参考（key）就激活该实现
    2. @Activate (group= “”)：指定 group 值，当 getActivateExtension 时指定 group 值时，匹配的将会被激活

步骤1.1：以 @Activate (value = “name”) 为例子

@Activate(value = "name")
public class Worker implements Person {
    static {
        System.out.println("Worker类被初始化");
    }
    @Override
    public void sayHi(URL url) {
        System.out.println("你好，我是玉树临风的工作人员");
    }
}
  • 注意：当前例子只对 Worker 添加了 @Activate(value = “name”)

步骤1.2：使用

@Test
public void testSPI() {

    ExtensionLoader<Person> loader = ExtensionLoader.getExtensionLoader(Person.class);

    String request = "http://www.baidu.com?";

为什么 Dubbo SPI 要扩展 JDK SPI

在 Dubbo 官方文档中，我们可以得知到它认为 JDK SPI 有如下缺点

  • 需要遍历所有的实现并将其实例化，还需要在循环中才能找到需要的实现。相当于说 API 不够友好
  • JDK SPI 的关联文件没有为实现命名，导致在程序中很难去准确的引用它们。相当于说 API 不够友好
  • 扩展如果依赖其他的扩展，做不到自动注入和装配。相当于说设计不够智能
  • 没提供类似于 Spring 的 IOC 和 AOP 功能。这里其实重复了上面需求
  • 扩展很难和其他的框架集成，比如扩展里面依赖了一个Spring bean，原生的 JDK SPI 机制不支持。相当于说 JDK SPI 机制扩展性差

根据官方文档的描述，总得来说 Dubbo 认为原生的 JDK SPI 机制存在以下问题：

  • API 不够友好
  • 设计不够智能
  • 可扩展性差

ExtensionLoader 源码浅析

普通模式

getOrDefaultExtension(String name) 方法

// 根据key获取实现，如果匹配不到则返回默认实现
public T getOrDefaultExtension(String name) {
    // containsExtension(name)：会先加载SPI扩展配置文件，并判断key值name是否存在于加载内容中
    // 如果返回true，则直接调用 getExtension(name) 获取，否者加载默认值(既@SPI中的value值)
    return containsExtension(name) ? getExtension(name) : getDefaultExtension();
}

containsExtension(name) 方法

// 判断key值name是否存在于加载内容中
private boolean containsExtension(String name) {
    return getExtensionClasses().containsKey(name);
}

getExtensionClasses() 方法

// 获取SPI扩展实现
private Map<String, Class<?>> getExtensionClasses() {
    // 1. 先从缓存中获取，没有进行加载
    Map<String, Class<?>> classes = cachedClasses.get();
    // 先判断缓存中是否存在，避免无谓消耗
    if (classes == null) {
        synchronized (cachedClasses) {
            classes = cachedClasses.get(); // 
            if (classes == null) {
                // 加载关联文件
                // classes为扩展点实现的Class集合
                classes = loadExtensionClasses();
                // 将其保存到 Holder<Map<String, Class<?>>> cachedClasses 中缓存
                cachedClasses.set(classes);
            }
        }
    }
    return classes;
}

loadExtensionClasses() 方法

// 从SPI关联文件中加载扩展点实现
private Map<String, Class<?>> loadExtensionClasses() {
    // 从扩展点的@SPI注解中得到默认的key值，如果有的话
    cacheDefaultExtensionName(); 
    // 扩展点的实现
    Map<String, Class<?>> extensionClasses = new HashMap<>();

    // 通过Dubbo内置的加载策略来加载SPI关联文件
    for (LoadingStrategy strategy : strategies) {
        // 参数1：存放加载内容的容器
        // 参数2：加载目录
        // 参数3：扩展点名称
        // 参数4：类加载器
        // 参数5：是否开启覆盖
        // 参数6：查找时排除的包
        loadDirectory(extensionClasses, strategy.directory(), type.getName(), strategy.preferExtensionClassLoader(), strategy.overridden(), strategy.excludedPackages());
        loadDirectory(extensionClasses, strategy.directory(), type.getName().replace("org.apache", "com.alibaba"), strategy.preferExtensionClassLoader(), strategy.overridden(), strategy.excludedPackages());
    }

    return extensionClasses;
}

在 Dubbo 中内置的加载策略有以下这些：

  • DubboInternalLoadingStrategy：从 META-INF/dubbo/internal/ 中加载
  • DubboLoadingStrategy：从 META-INF/dubbo/ 中加载
  • ServicesLoadingStrategy：从 META-INF/services/ 中加载

也就是说，Dubbo SPI 的关联文件可以存放在以上文件路径下。当然，你也可以通过实现 LoadingStrategy 接口，并在 META-INF/services 目录下添加其关联文件来扩展 Dubbo 内置的加载策略，譬如：

public class MyLoadingStrategy implements LoadingStrategy {
    @Override
    public String directory() { // 自定义关联文件加载路径
        return "META-INF/my/";
    }
    @Override
    public boolean overridden() {
        return true;
    }
    @Override
    public int getPriority() { // 优先级别
        return NORMAL_PRIORITY;
    }
}

loadDirectory(…) 方法

// 根据加载策略到指定的位置加载关联文件
private void loadDirectory(Map<String, Class<?>> extensionClasses, String dir, String type,
                           boolean extensionLoaderClassLoaderFirst, boolean overridden, String... excludedPackages) {
    String fileName = dir + type; // 关联文件的全限定名称
    try {
        Enumeration<java.net.URL> urls = null;
        ClassLoader classLoader = findClassLoader();
        // ...
        if (urls == null || !urls.hasMoreElements()) {
            // 这里和JDK SPI一样，如果没有指定类加载的前提下就使用加载当前类的类加载器加载
            if (classLoader != null) {
                urls = classLoader.getResources(fileName); 
            } else {
                urls = ClassLoader.getSystemResources(fileName);
            }
        }
        if (urls != null) {
            while (urls.hasMoreElements()) {
                java.net.URL resourceURL = urls.nextElement();
                // 进行资源加载
                loadResource(extensionClasses, classLoader, resourceURL, overridden, excludedPackages);
            }
        }
    } 
    // ...
}

loadResource(…) 方法

// 解析关联文件，并将其实现Class添加到容器extensionClasses中保存
private void loadResource(Map<String, Class<?>> extensionClasses, ClassLoader classLoader,
                          java.net.URL resourceURL, boolean overridden, String... excludedPackages) {
    try {
        try (BufferedReader reader = new BufferedReader(new InputStreamReader(resourceURL.openStream(), StandardCharsets.UTF_8))) {
            String line;
            // 一次读取一行，一行的格式是：key=value
            while ((line = reader.readLine()) != null) { 
                final int ci = line.indexOf('#');
                if (ci >= 0) {
                    line = line.substring(0, ci);
                }
                line = line.trim();
                if (line.length() > 0) {
                    try {
                        String name = null;
                        int i = line.indexOf('=');
                        if (i > 0) {
                            name = line.substring(0, i).trim();
                            line = line.substring(i + 1).trim();
                        }
                        if (line.length() > 0 && !isExcluded(line, excludedPackages)) { // 加载实现，并初始化
                            loadClass(extensionClasses, resourceURL, Class.forName(line, true, classLoader), name, overridden);
                        }
                    }
                    // ...
                }
            }
        }
    }
    // ...
}

getExtension(String name) 方法

// 根据name来获取实现
public T getExtension(String name) {
    if (StringUtils.isEmpty(name)) {
        throw new IllegalArgumentException("Extension name == null");
    }
    if ("true".equals(name)) {
        return getDefaultExtension();
    }
    final Holder<Object> holder = getOrCreateHolder(name);
    Object instance = holder.get(); // Object为Map<String, Class<?>>
    if (instance == null) { // 如果找不到，则扫描关联文件
        synchronized (holder) {
            instance = holder.get();
            if (instance == null) {
                instance = createExtension(name);
                holder.set(instance);
            }
        }
    }
    return (T) instance;
}

getDefaultExtension() 方法

// 获取默认实现
public T getDefaultExtension() {
    // 扫描关联文件
    // 注意，该方法并不会重复加载，因为在当前例子中调用getOrDefaultExtension()时结果已经被缓存
    // 这里调用getExtensionClasses()是因为当前方法getDefaultExtension()是public修饰的
    // 其实getOrDefaultExtension()方法只是组合了getExtension(name)和getDefaultExtension()两个public方法而已
    getExtensionClasses(); 
    if (StringUtils.isBlank(cachedDefaultName) || "true".equals(cachedDefaultName)) {
        return null;
    }
    // 可以看到找到默认key后，实质最终也是用getExtension(name)查找的
    return getExtension(cachedDefaultName);
}

注：cachedDefaultName 的值是在 loadExtensionClasses() 中的 cacheDefaultExtensionName() 确认的，其源码如下

// 从@SPI注解中得到默认key值
private void cacheDefaultExtensionName() {
    // 获取@SPI注解
    final SPI defaultAnnotation = type.getAnnotation(SPI.class);
    // 没有指定注解
    if (defaultAnnotation == null) { 
        return;
    }
    // 获取@SPI注解中的value值
    String value = defaultAnnotation.value();
    // value值不为空
    if ((value = value.trim()).length() > 0) {
        String[] names = NAME_SEPARATOR.split(value);
        // value值本身是String类型
        // 这里的意思是value值字符串不能存在","分隔符
        if (names.length > 1) {
            throw new IllegalStateException("More than 1 default extension name on extension " + type.getName()
                                            + ": " + Arrays.toString(names));
        }
        if (names.length == 1) {
            cachedDefaultName = names[0];
        }
    }
}

private Class<?> createAdaptiveExtensionClass() {

    ClassLoader classLoader = findClassLoader();

    public void sayHi(org.apache.dubbo.common.URL arg0) {

        if (arg0 == null) throw new IllegalArgumentException("url == null");

        extension.sayHi(arg0);

injectExtension(T instance) 方法

// 给Adapter代理实例进行参数注入（IOC），如果有必要的话
// 如果你没忘记，Dubbo认为JDK SPI的缺点之一就是扩展性差，譬如无法注入Spring Bean
// 而该方法就是解决这一问题的
private T injectExtension(T instance) {
    if (objectFactory == null) {
        return instance;
    }
    try {
        // 获取代理的所有方法
        for (Method method : instance.getClass().getMethods()) {
            // 如果不是Setter，则跳过
            if (!isSetter(method)) {
                continue;
            }
            // 如果方法被@DisableInject注解，则跳过
            if (method.getAnnotation(DisableInject.class) != null) {
                continue;
            }
            // 如果参数是基本数据类型（包含装箱类），则跳过
            // 这里只获取第一个参数是因为，Setter正常情况下只有1个参数
            Class<?> pt = method.getParameterTypes()[0];
            if (ReflectUtils.isPrimitives(pt)) {
                continue;
            }
            try {
                // 根据属性Setter方法名称，返回字段的名称（既去除set前三个字符）
                String property = getSetterProperty(method);
                // 根据字段在实际扩展实现中得到对应的值
                Object object = objectFactory.getExtension(pt, property);
                if (object != null) {
                    // 将其注入到代理中
                    method.invoke(instance, object);
                }
            } catch (Exception e) {
                logger.error("Failed to inject via method " + method.getName()
                             + " of interface " + type.getName() + ": " + e.getMessage(), e);
            }
        }
    } catch (Exception e) {
        logger.error(e.getMessage(), e);
    }
    return instance;
}

激活模式

getActivateExtension(URL url, String[] values, String group) 方法

// 根据条件来激活扩展实现
// 参数1：URL条件信息
// 参数2：必须激活的实现
// 参数3：组的名称，以组的形式激活一个或多个实现
public List<T> getActivateExtension(URL url, String[] values, String group) {
    List<T> activateExtensions = new ArrayList<>();
    List<String> names = values == null ? new ArrayList<>(0) : asList(values); //
    if (!names.contains(REMOVE_VALUE_PREFIX + DEFAULT_KEY)) { // REMOVE_VALUE_PREFIX=-,DEFAULT_KEY=default

        // 加载SPI扩展实现
        getExtensionClasses();

        // 激活筛选逻辑
        for (Map.Entry<String, Object> entry : cachedActivates.entrySet()) {
            String name = entry.getKey(); // 实现的key值
            Object activate = entry.getValue(); // @Activate注解对象
            String[] activateGroup, activateValue; // @Activate注解中的值
            // 获取@Activate注解中的value值和group值
            if (activate instanceof Activate) { // 新注解
                activateGroup = ((Activate) activate).group(); // group值
                activateValue = ((Activate) activate).value(); // value值
            } else if (activate instanceof com.alibaba.dubbo.common.extension.Activate) { // 旧注解
                activateGroup = ((com.alibaba.dubbo.common.extension.Activate) activate).group();
                activateValue = ((com.alibaba.dubbo.common.extension.Activate) activate).value();
            } else {
                continue;
            }
            if (isMatchGroup(group, activateGroup) // 根据group参数激活，如果没有指定group则默认返回true
                && !names.contains(name)
                && !names.contains(REMOVE_VALUE_PREFIX + name)
                && isActive(activateValue, url)) { // 根据URL激活，如果没有指定URL则默认返回true
                // 加载实现并将其实例对象暂存到activateExtensions中
                activateExtensions.add(getExtension(name));
            }
        }
        activateExtensions.sort(ActivateComparator.COMPARATOR);
    }
    // 加载必须的实现
    List<T> loadedExtensions = new ArrayList<>();
    for (int i = 0; i < names.size(); i++) {
        String name = names.get(i);
        if (!name.startsWith(REMOVE_VALUE_PREFIX)
            && !names.contains(REMOVE_VALUE_PREFIX + name)) {
            if (DEFAULT_KEY.equals(name)) {
                if (!loadedExtensions.isEmpty()) {
                    activateExtensions.addAll(0, loadedExtensions);
                    loadedExtensions.clear();
                }
            } else {
                loadedExtensions.add(getExtension(name));
            }
        }
    }
    if (!loadedExtensions.isEmpty()) {
        activateExtensions.addAll(loadedExtensions);
    }
    // 返回激活了的实现实例集合
    return activateExtensions;
}

cacheActivateClass(Class<?> clazz, String name) 方法

// 缓存带有@Activate注解的实现
private void cacheActivateClass(Class<?> clazz, String name) {
    // 获取当前扩展类的@Activate注解
    Activate activate = clazz.getAnnotation(Activate.class);
    if (activate != null) {
        cachedActivates.put(name, activate);
    } else {
        // 如果新注解没有，则检查是否有用到旧注解，旧注解包名包含为alibaba，新注解为apache
        // support com.alibaba.dubbo.common.extension.Activate
        com.alibaba.dubbo.common.extension.Activate oldActivate = clazz.getAnnotation(com.alibaba.dubbo.common.extension.Activate.class);
        if (oldActivate != null) {
            cachedActivates.put(name, oldActivate);
        }
    }
}

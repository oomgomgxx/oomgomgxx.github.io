HttpURLConnection

HttpURLConnection是JDK内置的HTTP请求工具

private String inputContent(HttpURLConnection connection) throws IOException {
    BufferedInputStream bis = new BufferedInputStream(connection.getInputStream());
    byte[] buff = new byte[bis.available()];
    int len = 0;
    StringBuilder sb = new StringBuilder();
    while ((len=bis.read(buff)) != -1) {
        sb.append(new String(buff, 0, buff.length));
    }
    bis.close();
    return sb.toString();
}

Spring RestTemplate

基本介绍

  RestTemplate位于 org.springframework.web 包下，是Spring提供的HTTP请求包装类。默认采用JDK的HttpURLConnection来实现。除了 HttpURLConnection 之外还支持 Netty4、OkHttp 还用 HttpClient。采用那一种实现取决于实现是否存在类路径下。顺序是 HttpClient 、OkHttp 、HttpURLConnection，即如果没有添加 HttpClient 和 OkHttp 的依赖就会采用 HttpURLConnection 来实现。

public abstract class HttpAccessor {
    // 默认请求工厂，可以看到是URLConnection
    private ClientHttpRequestFactory requestFactory = new SimpleClientHttpRequestFactory();

    public void setRequestFactory(ClientHttpRequestFactory requestFactory) {
        this.requestFactory = requestFactory;
    }

    public ClientHttpRequestFactory getRequestFactory() {
        return this.requestFactory;
    }
    // 使用HttpRequestFactory返回ClientHttpRequest
    protected ClientHttpRequest createRequest(URI url, HttpMethod method) throws IOException {
        ClientHttpRequest request = getRequestFactory().createRequest(url, method);
        return request;
    }
}

InterceptingHttpAccessor

public abstract class InterceptingHttpAccessor extends HttpAccessor {
    private final List<ClientHttpRequestInterceptor> interceptors = new ArrayList<>();
    @Nullable
    private volatile ClientHttpRequestFactory interceptingRequestFactory;
    public void setInterceptors(List<ClientHttpRequestInterceptor> interceptors) {
        // Take getInterceptors() List as-is when passed in here
        if (this.interceptors != interceptors) {
            this.interceptors.clear();
            this.interceptors.addAll(interceptors);
            AnnotationAwareOrderComparator.sort(this.interceptors);
        }
    }
    public List<ClientHttpRequestInterceptor> getInterceptors() {
        return this.interceptors;
    }
    @Override
    public void setRequestFactory(ClientHttpRequestFactory requestFactory) {
        super.setRequestFactory(requestFactory);
        this.interceptingRequestFactory = null;
    }
    @Override
    public ClientHttpRequestFactory getRequestFactory() {
        List<ClientHttpRequestInterceptor> interceptors = getInterceptors();
        if (!CollectionUtils.isEmpty(interceptors)) {
            // 获取拦截器版请求工厂
            ClientHttpRequestFactory factory = this.interceptingRequestFactory;
            if (factory == null) {
                // 使用父类的请求工厂（默认URLConnection）和拥有的拦截器创建一个 拦截器版请求工厂
                factory = new InterceptingClientHttpRequestFactory(super.getRequestFactory(), interceptors);
                this.interceptingRequestFactory = factory;
            }
            return factory;
        }
        else {
            return super.getRequestFactory();
        }
    }
}

点进 ClientHttpRequestFactorySupplier

private static final Map<String, String> REQUEST_FACTORY_CANDIDATES;
static {
    // 使用LinkedHashMap设置加载优先级别
    Map<String, String> candidates = new LinkedHashMap<>();
    candidates.put("org.apache.http.client.HttpClient",
            "org.springframework.http.client.HttpComponentsClientHttpRequestFactory");
    candidates.put("okhttp3.OkHttpClient",
            "org.springframework.http.client.OkHttp3ClientHttpRequestFactory");
    REQUEST_FACTORY_CANDIDATES = Collections.unmodifiableMap(candidates);
}

@Override
public ClientHttpRequestFactory get() {
    for (Map.Entry<String, String> candidate : REQUEST_FACTORY_CANDIDATES
            .entrySet()) {
        ClassLoader classLoader = getClass().getClassLoader();
        if (ClassUtils.isPresent(candidate.getKey(), classLoader)) {
            Class<?> factoryClass = ClassUtils.resolveClassName(candidate.getValue(),
                    classLoader);
            return (ClientHttpRequestFactory) BeanUtils
                    .instantiateClass(factoryClass);
        }
    }
    // 如果类路径中没有HttpClient和OkHttp则使用URLConnection
    return new SimpleClientHttpRequestFactory();
}

OpenFeign

基本介绍

Feign本来是Spring Cloud Netflix开源项目的一部分，后来从项目中独立了出来成为一个通用的Java类库并改名为OpenFeign。OpenFeign是一个声明式、模板化的HTTP客户端抽象，支持Feign、JAX-RS、SpringMVC 等风格注解，而且内置 Ribbon 和 Hystrix 功能。OpenFeign 和 RestTemplate 一样，默认采用 URLConnection 进行HTTP通信，同时也支持 Apache HttpClient 和 OkHttp。

public RequestTemplate resolve(Map<String, ?> variables) {
    RequestTemplate resolved = RequestTemplate.from(this);

    // 设置URI
    StringBuilder uri = new StringBuilder();
    uri.append(this.uriTemplate.expand(variables));

    // 设置请求参数
    if (!this.queries.isEmpty()) {
      resolved.queries(Collections.emptyMap());
      StringBuilder query = new StringBuilder();
      Iterator<QueryTemplate> queryTemplates = this.queries.values().iterator();
      while (queryTemplates.hasNext()) {
        QueryTemplate queryTemplate = queryTemplates.next();
        String queryExpanded = queryTemplate.expand(variables);
        if (Util.isNotBlank(queryExpanded)) {
          query.append(queryExpanded);
          if (queryTemplates.hasNext()) {
            query.append("&");
          }
        }
      }
      String queryString = query.toString();
      if (!queryString.isEmpty()) {
        Matcher queryMatcher = QUERY_STRING_PATTERN.matcher(uri);
        if (queryMatcher.find()) {
          uri.append("&");
        } else {
          uri.append("?");
        }
        uri.append(queryString);
      }
    }
    // 将值设置到URI
    resolved.uri(uri.toString());

    // 设置请求头
    if (!this.headers.isEmpty()) {
      resolved.headers(Collections.emptyMap());
      for (HeaderTemplate headerTemplate : this.headers.values()) {
        String header = headerTemplate.expand(variables);
        if (!header.isEmpty()) {
          String headerValues = header.substring(header.indexOf(" ") + 1);
          if (!headerValues.isEmpty()) {
            resolved.header(headerTemplate.getName(), headerValues);
          }
        }
      }
    }

    resolved.body(this.body.expand(variables));
    resolved.resolved = true;
    return resolved;
}

进入 Client.Default

public interface Client {
    // 抽象方法 execute
    Response execute(Request request, Options options) throws IOException;

    // FeignClient默认实现
    class Default implements Client { 

        @Override
        public Response execute(Request request, Options options) throws IOException {
            HttpURLConnection connection = convertAndSend(request, options);
            return convertResponse(connection, request);
        }

        HttpURLConnection convertAndSend(Request request, Options options) throws IOException {

            // 创建HttpURLConnection实例
            final HttpURLConnection connection = (HttpURLConnection) new URL(request.url()).openConnection();

            // 设置请求头
            // ...

            // 发送请求
            if (request.requestBody().asBytes() != null) {
                connection.setDoOutput(true);
                OutputStream out = connection.getOutputStream();
                out.write(request.requestBody().asBytes());
                out.close();
            }
            return connection;
        }
    }
}

public class LoadBalancerFeignClient implements Client {
    private final Client delegate;
    public LoadBalancerFeignClient(Client delegate,/**/) {
        this.delegate = delegate;
    }
    public Client getDelegate() {
        return this.delegate;
    }
    // 获取负载均衡器，FeignLoadBalancer并不是直接实现ILoadBalancer，而是通过继承LoadBalancerContext来实现
    // LoadBalancerContext集成了ILoadBalancer
    private FeignLoadBalancer lbClient(String clientName) {
        return this.lbClientFactory.create(clientName);
    }

    @Override
    public Response execute(Request request, Request.Options options) throws IOException {
        // 获取URL内容
        URI asUri = URI.create(request.url());
        String clientName = asUri.getHost();
        URI uriWithoutHost = cleanUrl(request.url(), clientName);

        // 构造一个RibbonRequest，其中Request包含了FeignClient
        FeignLoadBalancer.RibbonRequest ribbonRequest = new FeignLoadBalancer.RibbonRequest(this.delegate, request, uriWithoutHost);
        IClientConfig requestConfig = getClientConfig(options, clientName);

        // 【重点】
        // 1. FeignClient会使用Ribon默认的负载均衡器ZoneAwareLoadBalancer，即采用最优zone加RoundRibbonRule进行负载均衡
        // 2. 得到负载均衡后的Server使用HTTP客户端发送请求并返回结果
        return lbClient(clientName).executeWithLoadBalancer(ribbonRequest, requestConfig).toResponse();
    }
}

基于上面代码，先察一下 FeignLoadBalancer

  • 可以看到 executeWithLoadBalancer 方法是 FeignLoadBalancer 继承 AbstractLoadBalancerAwareClient 得到的

所以，FeignLoadBalancer 的 AbstractLoadBalancerAwareClient 如下：

public T executeWithLoadBalancer(final S request, final IClientConfig requestConfig) throws ClientException {
    LoadBalancerCommand<T> command = buildLoadBalancerCommand(request, requestConfig);
    // 【重点】负载均衡在LoadBalancerCommand的submit方法中完成。这里是lambda表达式，实际还需要进入submit查看
    return command.submit(
        new ServerOperation<T>() {
            @Override
            public Observable<T> call(Server server) {
                URI finalUri = reconstructURIWithServer(server, request.getUri());
                S requestForServer = (S) request.replaceUri(finalUri);
                // 调用FeignLoadBalancer的execute()方法
                return Observable.just(AbstractLoadBalancerAwareClient.this.execute(requestForServer, requestConfig));
            }
        })
        .toBlocking()
        .single();
}

public Observable<T> submit(final ServerOperation<T> operation) {
    // 如果没有指定Server的话就使用selectServer()进行负载均衡操作
    Observable<T> o = (server == null ? selectServer() : Observable.just(server)).concatMap(/**/);
    //...

}

private Observable<Server> selectServer() {
    return Observable.create(new OnSubscribe<Server>() {
        @Override
        public void call(Subscriber<? super Server> next) {
            // 先前提到FeignLoadBalancer并没有实现ILoadBalancer接口
            // 而是通过LoadBalancerContext继承ILoadBalancer实现负载均衡操作
            Server server = loadBalancerContext.getServerFromLoadBalancer(loadBalancerURI, loadBalancerKey);   
            next.onNext(server);
            next.onCompleted();
        }
    });
}

LoadBalancerContext 的 getServerFromLoadBalancer方法

public Server getServerFromLoadBalancer(@Nullable URI original, @Nullable Object loadBalancerKey) throws ClientException {
    String host = null;
    int port = -1;
    if (original != null) host = original.getHost();
    if (original != null) {
        Pair<String, Integer> schemeAndPort = deriveSchemeAndPortFromPartialUri(original);        
        port = schemeAndPort.second();
    }

    // 获取负载均衡器
    ILoadBalancer lb = getLoadBalancer();

    if (host == null) {
        if (lb != null){
            // 获取负载均衡后的Server
            Server svc = lb.chooseServer(loadBalancerKey);
            host = svc.getHost();
            return svc;
        } 
    }
    return new Server(host, port);
}

  • 以上是 SynchronousMethodHandler 的一段代码，可以看到在 Feign 中采用功能开关/模式开关的操作来控制日志的数据
  • Feign只输出 DEBUG 级别的日志，Feign Logger中的 模式Level 有以下几种：
    • NONE：不记录任何日志（默认级别）
    • BASIC：记录请求方法、URL、响应状态码以及执行的时间
    • HEADERS：BASIC + header 信息
    • FULL：记录请求和响应的 header、body、metadata等信息

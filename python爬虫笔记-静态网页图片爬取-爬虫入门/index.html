# =========== 图片保存相关 ===========
import os

dirName = 'C:/Users/admin/Desktop/py_save'

# 创建文件夹
def mkdir(path):
    path = path.strip()
    isExists = os.path.exists(path)
    if not isExists:
        os.makedirs(path)
        print('文件夹创建成功')
    else:
        print(path, '文件夹已经存在了，不再创建')

# 创建文件夹
mkdir(dirName)

# 保存图片
def saveImg(requests, url, dirName, fileName):
    img = requests.get(url)
    file_name = dirName + '/' + fileName + '.png'
    f = open(file_name, 'ab')
    f.write(img.content)
    print(file_name, '文件保存成功！')
    f.close()

# =========== 去除https://unsplash.com/t/wallpapers中图片地址后面的参数 ===========

reqUrl = 'https://unsplash.com/t/wallpapers'
text = requests.get(reqUrl).text
soup = BeautifulSoup(text, 'lxml')

import uuid

#得到所有的<a>标签
find_all = soup.find_all('img')
for img in find_all:

    # 文件名称
    uuid_ = uuid.uuid1()

    if img.get('src'):
        # 将后面的参数去掉，即?后面的内容
        url = img.get('src').split('?')[0]
        saveImg(requests, url, dirName, str(uuid_))

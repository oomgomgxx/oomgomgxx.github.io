公共配置中心介绍

基本特点

  • 提供集中化外部(公共)配置
  • 支持配置推送
  • 提供配置加密功能
  • 支持配置的环境切换

常用配置中心

  • SpringCloud Config
  • HashiCorp Consul
  • 携程 Apollo
  • 阿里巴巴 Nacos
  • ZooKeeper

bootstrap.yml和application.yml区别

  • bootstrap.yml 加载顺序优先于 application.yml
  • 因为 application.yml 在 bootstrap.yml 后，所以前者配置可以覆盖后者配置
  • 如果使用了配置中心，则配置中心的相关配置及其依赖到的配置都应该在 bootstrap.yml 中配置

SpringCloud Config

基本介绍

  • SpringCloud Config主要分为两个角色：Config Server、Config Client
  • SpringCloud Config默认采用Git库来存储外部配置。除此之外还支持Subversion、FileSystem、CredHub、JDBC等等
  • Config Client 会在微服务启动是的环境初始化阶段向 Config Server 获取外部配置并添加到自己的环境中

快速入门

Config Server

  • 规范的做法是 Config Server 应该是一个独立的微服务

到GitHub或者Gitlab中创建存储配置的库

  • 步骤1：创建存储配置的库

  • 步骤2：上传微服务外置配置内容文件

    各个文件内容如下（实际开发根据自己的需求来抽取配置）：

    # msa-config-client-default.yml

  desc: default

  desc: dev
  qq: 497870337

# msa-config-client-test.yml
info:
  desc: test

# msa-config-client-pro.yml
info:
  desc: pro

    关于外置配置文件的命名规则：

    • 格式：微服务名称-环境。文件可以是yaml或者properties
    • 微服务名称可以表示为application。环境则可以表示为profile
    • 例如，msa-config-client-test.yml。其中 application 就是 msa-config-client，而 profile 则是 test
    • 需要强调的是，微服务中的 spring.application.name 必须和外置配置文件名称的application部分对应，否则无法获取到外置配置。换句话就是 spring.application.name 必须和 外置配置文件名称的 application 一致

    private ResourceRepository resourceRepository;

    private UrlPathHelper helper = new UrlPathHelper();

    @RequestMapping("/{name}/{profile}/{label}/**")
    public String retrieve(@PathVariable String name, @PathVariable String profile,
            @PathVariable String label, ServletWebRequest request,
            @RequestParam(defaultValue = "true") boolean resolvePlaceholders)
            throws IOException {
        String path = getFilePath(request, name, profile, label);
        return retrieve(request, name, profile, label, path, resolvePlaceholders);
    }
    @RequestMapping(value = "/{name}/{profile}/**", params = "useDefaultLabel")
    public String retrieve(@PathVariable String name, @PathVariable String profile,
            ServletWebRequest request,
            @RequestParam(defaultValue = "true") boolean resolvePlaceholders)
            throws IOException {
        String path = getFilePath(request, name, profile, null);
        return retrieve(request, name, profile, null, path, resolvePlaceholders);
    }

    private String getFilePath(ServletWebRequest request, String name, String profile,
            String label) {
        String stem;
        if (label != null) {
            stem = String.format("/%s/%s/%s/", name, profile, label);
        }
        else {
            stem = String.format("/%s/%s/", name, profile);
        }
        String path = this.helper.getPathWithinApplication(request.getRequest());
        path = path.substring(path.indexOf(stem) + stem.length());
        return path;
    }

    @RequestMapping(value = "/{name}/{profile}/{label}/**", produces = MediaType.APPLICATION_OCTET_STREAM_VALUE)
    public byte[] binary(@PathVariable String name, @PathVariable String profile,
            @PathVariable String label, ServletWebRequest request) throws IOException {

    // 重试操作：需要spring retry和spring aop依赖支持
    @ConditionalOnProperty("spring.cloud.config.fail-fast")
    @ConditionalOnClass({ Retryable.class, Aspect.class, AopAutoConfiguration.class })
    @Configuration
    @EnableRetry(proxyTargetClass = true)
    @Import(AopAutoConfiguration.class)
    @EnableConfigurationProperties(RetryProperties.class)
    protected static class RetryConfiguration {
        @Bean
        @ConditionalOnMissingBean(name = "configServerRetryInterceptor")
        public RetryOperationsInterceptor configServerRetryInterceptor(
                RetryProperties properties) {
            return RetryInterceptorBuilder.stateless()
                    .backOffOptions(properties.getInitialInterval(),
                            properties.getMultiplier(), properties.getMaxInterval())
                    .maxAttempts(properties.getMaxAttempts()).build();
        }

    }

进入 ConfigServicePropertySourceLocator

@Order(0) // 最高优先级别(Order值最小优先级别越高)
public class ConfigServicePropertySourceLocator implements PropertySourceLocator {
    private RestTemplate restTemplate;
    private ConfigClientProperties defaultProperties;
    // ...

    @Override
    @Retryable(interceptor = "configServerRetryInterceptor")
    public org.springframework.core.env.PropertySource<?> locate(
            org.springframework.core.env.Environment environment) {

        ConfigClientProperties properties = this.defaultProperties.override(environment);
        CompositePropertySource composite = new CompositePropertySource("configService");
        RestTemplate restTemplate = this.restTemplate == null? getSecureRestTemplate(properties) : this.restTemplate;
        try {
            // ...
            for (String label : labels) {
                // 获取远程配置
                Environment result = getRemoteEnvironment(restTemplate, properties,label.trim(), state);
                if (result != null) {
                    if (result.getPropertySources() != null) {
                        for (PropertySource source : result.getPropertySources()) {
                            @SuppressWarnings("unchecked")
                            Map<String, Object> map = (Map<String, Object>) source
                                    .getSource();
                            composite.addPropertySource(
                                    new MapPropertySource(source.getName(), map));
                        }
                    }

                    if (StringUtils.hasText(result.getState())
                            || StringUtils.hasText(result.getVersion())) {
                        HashMap<String, Object> map = new HashMap<>();
                        putValue(map, "config.client.state", result.getState());
                        putValue(map, "config.client.version", result.getVersion());
                        composite.addFirstPropertySource(new MapPropertySource("configClient", map));
                    }
                    // 返回远程配置map(CompositePropertySource)
                    return composite;
                }
            }
        }
        // ...
    }

    private Environment getRemoteEnvironment(RestTemplate restTemplate,
            ConfigClientProperties properties, String label, String state) {
        // 请求格式
        String path = "/{name}/{profile}";
        String name = properties.getName();
        String profile = properties.getProfile();
        String token = properties.getToken();
        int noOfUrls = properties.getUri().length;

        if (noOfUrls > 1) {
            logger.info("Multiple Config Server Urls found listed.");
        }

        Object[] args = new String[] { name, profile };
        if (StringUtils.hasText(label)) {
            if (label.contains("/")) {
                label = label.replace("/", "(_)");
            }
            args = new String[] { name, profile, label };
            path = path + "/{label}";
        }

        ResponseEntity<Environment> response = null;
        for (int i = 0; i < noOfUrls; i++) {
            Credentials credentials = properties.getCredentials(i);
            String uri = credentials.getUri();
            String username = credentials.getUsername();
            String password = credentials.getPassword();

            // 构建请求头
            HttpHeaders headers = new HttpHeaders();
            // 添加请求权限
            addAuthorizationToken(properties, headers, username, password);
            if (StringUtils.hasText(token)) {
                headers.add(TOKEN_HEADER, token);
            }
            if (StringUtils.hasText(state) && properties.isSendState()) {
                headers.add(STATE_HEADER, state);
            }
            headers.setAccept(Collections.singletonList(MediaType.APPLICATION_JSON));
            final HttpEntity<Void> entity = new HttpEntity<>((Void) null, headers);

            // 发送请求
            response = restTemplate.exchange(uri + path, HttpMethod.GET, entity,Environment.class, args);
            Environment result = response.getBody();
            return result;
        }

        return null;
    }

    private void addAuthorizationToken(ConfigClientProperties configClientProperties,
            HttpHeaders httpHeaders, String username, String password) {
        String authorization = configClientProperties.getHeaders().get(AUTHORIZATION);
        if (password != null) {
            byte[] token = Base64Utils.encode((username + ":" + password).getBytes());
            httpHeaders.add("Authorization", "Basic " + new String(token));
        }
        else if (authorization != null) {
            httpHeaders.add("Authorization", authorization);
        }

    }
}
SpringBoot是如何刷新配置？

SpringBoot Actuator 的 refresh 端点

在快如入门中我们已经尝试过当配置被修改后如何刷新配置，就是使用 SpringBoot Actuator 提供的端点操作 /refresh 来实现。以下为 refresh 端点源码：

@Endpoint(id = "refresh")
public class RefreshEndpoint {
    // 上下文刷新器
    private ContextRefresher contextRefresher;
    public RefreshEndpoint(ContextRefresher contextRefresher) {
        this.contextRefresher = contextRefresher;
    }

    @WriteOperation
    public Collection<String> refresh() {
        // 刷新配置操作
        Set<String> keys = this.contextRefresher.refresh();
        return keys;
    }
}
  • 可以看到，当请求 /refresh端点的时候实质就是使用 ContextRefresher 来刷新配置的
  • 现在得知的信息是 ContextRefresher 可以刷新 SpringBoot 的配置。那么ContextRefresher 是怎么创建的呢？

现在我们知道了刷新操作实质就是 ContextRefresher.refresh() 。也了解到 RefreshScope 负责@RefreshScope Bean刷新。那么我们进入 ContextRefresher 来研究一下源码

public class ContextRefresher {
    private static final String REFRESH_ARGS_PROPERTY_SOURCE = "refreshArgs";

    // 默认配置源
    private static final String[] DEFAULT_PROPERTY_SOURCES = new String[] {
            CommandLinePropertySource.COMMAND_LINE_PROPERTY_SOURCE_NAME, // commandLineArgs
            "defaultProperties" };
    // 标准配置源
    private Set<String> standardSources = new HashSet<>(
            Arrays.asList(StandardEnvironment.SYSTEM_PROPERTIES_PROPERTY_SOURCE_NAME, // systemProperties
                    StandardEnvironment.SYSTEM_ENVIRONMENT_PROPERTY_SOURCE_NAME, // systemEnvironment
                    StandardServletEnvironment.JNDI_PROPERTY_SOURCE_NAME, // jndiProperties
                    StandardServletEnvironment.SERVLET_CONFIG_PROPERTY_SOURCE_NAME, // servletConfigInitParams
                    StandardServletEnvironment.SERVLET_CONTEXT_PROPERTY_SOURCE_NAME, // servletContextInitParams
                    "configurationProperties"));

    private ConfigurableApplicationContext context;
    private RefreshScope scope;

    public ContextRefresher(ConfigurableApplicationContext context, RefreshScope scope) {
        this.context = context;
        this.scope = scope;
    }

    // 【重】刷新环境配置
    public synchronized Set<String> refresh() {
        // 刷新环境，并返回改动配置
        Set<String> keys = refreshEnvironment();
        // 清除上下文对Scope为Refresh的Bean的缓存
        // 这里的上下文指的是StandardScopeCache和ThreadLocalScopeCache
        this.scope.refreshAll();
        return keys;
    }
    public synchronized Set<String> refreshEnvironment() {
        // 刷新前的配置
        Map<String, Object> before = extract(this.context.getEnvironment().getPropertySources());
        // 【重】刷新配置核心操作
        addConfigFilesToEnvironment();
        // 刷新后的配置（只有新增或新修改的配置）
        Set<String> keys = changes(before,extract(this.context.getEnvironment().getPropertySources())).keySet();
        // 在当前上下文中发布一个环境修改事件
        this.context.publishEvent(new EnvironmentChangeEvent(this.context, keys));
        return keys;
    }

    ConfigurableApplicationContext addConfigFilesToEnvironment() {
        ConfigurableApplicationContext capture = null;
        try {
            // 获取当前环境（只包含DEFAULT_PROPERTY_SOURCES）
            StandardEnvironment environment = copyEnvironment(this.context.getEnvironment());

            /**** 核心思想:创建一个新的SpringBoot来重新加载一遍配置(从另外角度来看这是个重量级操作) ****/

            // 获取一个SpringBoot应用构建者
            SpringApplicationBuilder builder = new SpringApplicationBuilder(Empty.class)
                    .bannerMode(Mode.OFF) // 不打印Banner
                    .web(WebApplicationType.NONE) // 不是一个Web应用，即无需启动Web容器
                    .environment(environment); // 当前环境（只包含DEFAULT_PROPERTY_SOURCES）作为参数
            builder.application().setListeners(Arrays.asList(new BootstrapApplicationListener(),
                            new ConfigFileApplicationListener()));
            // 启动SpringBoot
            capture = builder.run(); 

            if (environment.getPropertySources().contains(REFRESH_ARGS_PROPERTY_SOURCE)) {
                environment.getPropertySources().remove(REFRESH_ARGS_PROPERTY_SOURCE);
            }

            // 当前程序所处的环境（主环境）
            MutablePropertySources target = this.context.getEnvironment().getPropertySources();
            String targetName = null;
            // 将SpringBoot加载到的内容添加到主环境
            for (PropertySource<?> source : environment.getPropertySources()) {
                String name = source.getName();
                if (target.contains(name)) {
                    targetName = name;
                }
                if (!this.standardSources.contains(name)) {
                    if (target.contains(name)) {
                        // 主环境包含该配置则直接替换值
                        target.replace(name, source);
                    }
                    // 如果主环境中没有该配置则添加
                    else {
                        if (targetName != null) {
                            target.addAfter(targetName, source);
                        }
                        else {
                            // targetName was null so we are at the start of the list
                            target.addFirst(source);
                            targetName = name;
                        }
                    }
                }
            }
        }
        return capture;
    }
}

org.springframework.cloud.context.scope.refresh.RefreshScope#refreshAll

// 大概意思是，处理当前范围的Bean，让其下次操作时达到刷新的效果
@ManagedOperation(description = "Dispose of the current instance of all beans in this scope and force a refresh on next method execution.")
public void refreshAll() {
    super.destroy(); // 清楚 ScopeCache
    // 发布刷新域已成功刷新事件，以广播给监听者做对应处理
    this.context.publishEvent(new RefreshScopeRefreshedEvent());
}
  • ScopeCache的意义在于当前操作上下文。例如用户正在操作系统，但在这期间刷新了配置，那么通过清楚上下文的ScopeCache，用户在下一个操作就能够感知到
  • 关于ScopeCache 参考：org.springframework.cloud.context.scope.GenericScope

RefreshScopeRefreshedEvent 的监听者 EurekaClientConfigurationRefresher

@Configuration
@ConditionalOnClass(RefreshScopeRefreshedEvent.class)
protected static class EurekaClientConfigurationRefresher
        implements ApplicationListener<RefreshScopeRefreshedEvent> {

    @Autowired(required = false)
    private EurekaClient eurekaClient;
    @Autowired(required = false)
    private EurekaAutoServiceRegistration autoRegistration;

    public void onApplicationEvent(RefreshScopeRefreshedEvent event) {
        // 如果还没有创建EurekaClient，则将强制创建
        // 以确保刷新事件后顺利重新注册客户端
        if (eurekaClient != null) 
            eurekaClient.getApplications();
        }
        if (autoRegistration != null) {
            // 从注册中心下线服务
            this.autoRegistration.stop();
            // 注册服务到注册中心
            this.autoRegistration.start();
        }
    }
}
  • 重新注册服务的意义在于，预防当前 Discovery Client 注册在注册中心的元数据被更改了

配置刷新总结：

  • 如果一个Bean依赖环境中的配置，应该使用 @Scope('refresh') 或 @RefreshScope 标注，只有这样Bean才可以感知到配置刷新。原因是 SpringCloud 会将被 @RefreshScope 标注的Bean打包成 BeanLifecycleWrapper 存放到 ScopeCache 中（ConcurrentHashMap）作为操作上限文使用的Bean，当刷新操作触发时就会将 ScopeCache 的内容删除，这样一来下次使用Bean时就会重新打包 BeanLifecycleWrapper 以达到刷新效果
  • @RefreshScope不一定要使用了配合中心才可以使用。手动修改了配置都可以使用端点来刷新配置
  • 如果想手动实现刷新，则可以参考 RefreshEndpoint 使用 ContextRefresher 来实现

    @Configuration

 * 官方注释：

@SuppressWarnings("serial")
public class ThreadPoolTaskScheduler extends ExecutorConfigurationSupport
        implements AsyncListenableTaskExecutor, SchedulingTaskExecutor, TaskScheduler {
    // 单线程
    private volatile int poolSize = 1;

    @Override
    public ScheduledFuture<?> scheduleWithFixedDelay(Runnable task, long delay) {
        ScheduledExecutorService executor = getScheduledExecutor();
        // 参数1：任务
        // 参数2：第一次执行的延时时间
        // 参数3：经过第一次后的延时时间
        // 参数4：延时时间的单位
        return executor.scheduleWithFixedDelay(errorHandlingTask(task, true), 0, delay, TimeUnit.MILLISECONDS);

public class ConfigWatch implements ApplicationEventPublisherAware, SmartLifecycle {

    private final ConsulConfigProperties properties;
    private final ConsulClient consul;
    private final TaskScheduler taskScheduler;
    private final AtomicBoolean running = new AtomicBoolean(false);
    private ApplicationEventPublisher publisher;

    @Override
    public void start() {
        // 将任务切换为已开启
        if (this.running.compareAndSet(false, true)) {
            this.watchFuture = this.taskScheduler.scheduleWithFixedDelay(
                    this::watchConfigKeyValues,  // 任务
                    this.properties.getWatch().getDelay()); // 延时值从配置获取
        }
    }
    // 任务
    @Timed("consul.watch-config-keys")
    public void watchConfigKeyValues() {
        // 任务是否处于开启状态
        if (this.running.get()) { 
            for (String context : this.consulIndexes.keySet()) {
                //...
                // 获取Consul上的KV值
                Response<List<GetValue>> response = this.consul.getKVValues(context,
                        aclToken,
                        new QueryParams(this.properties.getWatch().getWaitTime(),
                                currentIndex));

                if (response.getValue() != null && !response.getValue().isEmpty()) {
                    Long newIndex = response.getConsulIndex();
                    // 是否有配置？
                    if (newIndex != null && !newIndex.equals(currentIndex)) {
                        // 是不是新值？
                        if (!this.consulIndexes.containsValue(newIndex) && !currentIndex.equals(-1L)) {
                            //【重】发布刷新配置事件，要求RefreshEvent的监听者进行刷新操作
                            RefreshEventData data = new RefreshEventData(context,currentIndex, newIndex);
                            this.publisher.publishEvent(new RefreshEvent(this, data, data.toString()));
                        }
                    }
                }
            }
        }
    }    
}

public class RefreshEventListener implements SmartApplicationListener {
    private ContextRefresher refresh;
    private AtomicBoolean ready = new AtomicBoolean(false);
    public RefreshEventListener(ContextRefresher refresh) {
        this.refresh = refresh;
    }

    @Override
    public void onApplicationEvent(ApplicationEvent event) {
        //...
        handle((RefreshEvent) event);
    }

    public void handle(RefreshEvent event) {
        if (this.ready.get()) {
            log.debug("Event received " + event.getEventDesc());
            // 【重】使用ContextRefresher刷新配置
            // 即新建一个SpringBoot加载配置然后设置到当前环境中
            // 并主动下线注册中心的服务然后再重新注册
            Set<String> keys = this.refresh.refresh();
            log.info("Refresh keys changed: " + keys);
        }
    }
}

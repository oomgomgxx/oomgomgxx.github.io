认识Minio

 2018-09-07   作者: TanDi

minio

官方文档：https://docs.minio.io/

是什么？

  • Minio是Apache License v2.0下发布的对象存储服务器
  • 兼容Amazon S3云存储服务
  • 适合存储非结构化数据，如照片，视频，日志文件，备份和容器/ VM镜像

下载安装（centos7）

  1. 手动下载ftp传入或wget下载
  2. 将minio保存到/bin下，并授权后，tab键就可以补全
[root@localhost bin]# wget https://dl.minio.io/server/minio/release/linux-ppc64le/minio
[root@localhost bin]# chmod +x ./*

package com.td.utils;

import io.minio.MinioClient;
import io.minio.errors.MinioException;

public class FileUploader {
    private static String MINIO_URL = "http://192.168.4.157:9000";
    private static String ACCESS_KEY = "NV2JLV2HFKMAQRJCNHK0";
    private static String SECRET_KEY = "FY+YsoARfp6vIVeX8EoIIpBV+KB7XCjwyTbLWA8I";
    public static void main(String[] args) throws Exception {
        try {
            // Create a minioClient with the Minio Server name, Port, Access key and Secret key.
            MinioClient minioClient = new MinioClient(MINIO_URL, ACCESS_KEY, SECRET_KEY);

            // Check if the bucket already exists.
            boolean isExist = minioClient.bucketExists("image");
            if(isExist) {
                System.out.println("Bucket already exists.");
            } else {
                // Make a new bucket called asiatrip to hold a zip file of photos.
                minioClient.makeBucket("image");
            }

            // Upload the zip file to the bucket with putObject
            minioClient.putObject("image","xxx01.jpg", "C:\\Users\\admin\\Pictures\\xxx01.jpg");
            System.out.println("上传成功");
        } catch(MinioException e) {
            System.out.println("Error occurred: " + e);
        }
    }
}

下载
package com.td.utils;

import io.minio.MinioClient;
import io.minio.errors.MinioException;

import java.io.InputStream;
import java.nio.file.Files;
import java.nio.file.Paths;

public class FileUploader {
    private static String MINIO_URL = "http://192.168.4.157:9000";
    private static String ACCESS_KEY = "NV2JLV2HFKMAQRJCNHK0";
    private static String SECRET_KEY = "FY+YsoARfp6vIVeX8EoIIpBV+KB7XCjwyTbLWA8I";

    private static String FILE_NAME = "xxx01.jpg";

    public static void main(String[] args) throws Exception {
        try {
            // Create a minioClient with the Minio Server name, Port, Access key and Secret key.
            MinioClient minioClient = new MinioClient(MINIO_URL, ACCESS_KEY, SECRET_KEY);

            // Check if the bucket already exists.
            boolean isExist = minioClient.bucketExists("image");
            if(isExist) {
                System.out.println("Bucket already exists.");
            } else {
                // Make a new bucket called asiatrip to hold a zip file of photos.
                minioClient.makeBucket("image");
            }

            // Upload the zip file to the bucket with putObject
            minioClient.putObject("image", "xxx01", "C:\\Users\\admin\\Pictures\\" + FILE_NAME);
            System.out.println("上传成功");

            InputStream in = minioClient.getObject("image", FILE_NAME);
            Files.copy(in, Paths.get("C:\\Users\\admin\\Desktop", FILE_NAME));
            System.out.println("保存成功");
        } catch(MinioException e) {
            System.out.println("Error occurred: " + e);
        }
    }
}

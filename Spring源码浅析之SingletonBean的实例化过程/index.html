类之间的关系

关于BeanFactory和ApplicationContext的区别

  • BeanFactory的主要功能

    • 依赖查找
    • 依赖注入
    • Bean管理（注册、注销）

  • Application的主要功能

    • 资源管理（ResourceLoader）
    • 事件传播（ApplicationEventPublisher）
    • 国际化（MessageSource）

  • 之间的关系如下，可以看到它们之间是组合的关系。Application是BeanFactory的扩展

refresh() - 刷新启动容器

该方法继承自AbstractApplicationContext类

@Override
public void refresh() throws BeansException, IllegalStateException {
    synchronized (this.startupShutdownMonitor) {
        // 预初始化
        // PropertySources
        // 1. OriginTrackMapPropertySource：加载SpringBoot的配置文件（application.yml）
        // 2. OriginAwareSystemEnvironmentPropertySource：加载系统环境变量
        // 3. PropertiesPropertySource：加载系统属性
        prepareRefresh();

        // 创建基础容器（ConfigurableListableBeanFactory）
        // 注意：这里包含了设置父容器（BeanFactory）的逻辑
        ConfigurableListableBeanFactory beanFactory = obtainFreshBeanFactory();

        // 初始化容器，譬如添加内部Bean
        prepareBeanFactory(beanFactory);

        try {
            // 这是一个模板方法，可以针对性地做些特殊处理
            // 此时已经加载了所有JavaConfig（或XMl）的BeanDefinition
            postProcessBeanFactory(beanFactory);

            // 注册并执行ConfigurationClassPostProcessor
            // 注意：这里会解析出配置类的BeanDefinition（譬如@Configuration）
            invokeBeanFactoryPostProcessors(beanFactory);

            // 创建并注册BeanPostProcessor
            // 在注册时会通过BeanFactory.getBean()获取或创建（默认反射创建）
            // 1. 手动实现的BeanPostProcessors
            // 2. AutowiredAnnotationBeanPostProcessor：用于处理@Autowired、@Value、@Inject
            // 3. CommonAnnotationBeanPostProcessor：用于处理@Resource、@PreConstruct、@PostDestroy
            // 4. PersistenceBeanPostProcessor(JPA包)：用于处理@PersistenceContext
            registerBeanPostProcessors(beanFactory);

            // 国际化相关初始化
            initMessageSource();

            // 事件相关初始化
            initApplicationEventMulticaster();

            // 这是一个模板方法，可以针对性地做些特殊处理
            // 此时bean还没有实例化
            onRefresh();

            // 注册当前上限文中的事件监听者
            registerListeners();

            // 【注】实例化非延时加载外的单例Bean
            // 默认实例化顺序的规则如下(可用@Order或@DependsOn修改)：
            // 1. 先实例化配置类
            // 2. 定义类配置类上的注解，譬如@EnableWebMvc、@Import之类
            // 3. 按照配置类的定义从上到下实
            // 4. 按需实例化。譬如优先级较低的bean被以上bean依赖了，就会提前实例化
            finishBeanFactoryInitialization(beanFactory);

            // 发布上下文刷启动事件
            finishRefresh();
        }

        catch (BeansException ex) {
            // 异常处理...
        }

        finally {
            // 容器启动完成后清理在启动期间产生的元数据
            resetCommonCaches();
        }
    }
}

beanFactory.preInstantiateSingletons()

public void preInstantiateSingletons() throws BeansException {

    // 返回通过资源解析器解析配置（xml、class）得到的bean的名字
    List<String> beanNames = new ArrayList<>(this.beanDefinitionNames);

    // Trigger initialization of all non-lazy singleton beans...
    for (String beanName : beanNames) {
        RootBeanDefinition bd = getMergedLocalBeanDefinition(beanName);
        // 实例化条件：
        // 1、非抽象 2、非延时加载 3、单例
        if (!bd.isAbstract() && bd.isSingleton() && !bd.isLazyInit()) {   

            // 情况1：Bean实例通过FactoryBean接口产生
            if (isFactoryBean(beanName)) { 
                Object bean = getBean(FACTORY_BEAN_PREFIX + beanName);
                if (bean instanceof FactoryBean) { // 譬如 JPA 中 Repository 接口
                    final FactoryBean<?> factory = (FactoryBean<?>) bean;    
                    // ...
                    if (isEagerInit) {
                        getBean(beanName);
                    }
                }
            }

            // 情况2：其他
            else {
                getBean(beanName);
            }
        }
    }

AbstractBeanFactory#doGetBean 

protected <T> T doGetBean(final String name, @Nullable final Class<T> requiredType,
                          @Nullable final Object[] args, boolean typeCheckOnly) throws BeansException {

    final String beanName = transformedBeanName(name);
    Object bean;

    // 先到容器中检查是有存在该bean（存在表示已经创建并注册）
    Object sharedInstance = getSingleton(beanName);
    if (sharedInstance != null && args == null) {
        bean = getObjectForBeanInstance(sharedInstance, name, beanName, null);
    }
    else {
        // 如果有父容器，且当前容器并没有该Bean的定义，则尝试在父容器中获取
        BeanFactory parentBeanFactory = getParentBeanFactory();
        if (parentBeanFactory != null && !containsBeanDefinition(beanName)) {
            // ...
        }

        try {
            final RootBeanDefinition mbd = getMergedLocalBeanDefinition(beanName);
            checkMergedBeanDefinition(mbd, beanName, args);

            // 实例化Bean之前，先确保其依赖已经被实例化（@DependsOn）
            String[] dependsOn = mbd.getDependsOn();
            if (dependsOn != null) {
                for (String dep : dependsOn) {
                    registerDependentBean(dep, beanName);
                    getBean(dep);
                }
            }
            // 情况1：创建Singleton Bean
            if (mbd.isSingleton()) {
                sharedInstance = getSingleton(beanName, () -> {
                    return createBean(beanName, mbd, args); // 创建Bean
                });
                bean = getObjectForBeanInstance(sharedInstance, name, beanName, mbd);
            }
            // 情况2：创建Prototype Bean（当前不会被调用）
            else if (mbd.isPrototype()) {        
                Object prototypeInstance = null;
                try {
                    beforePrototypeCreation(beanName);
                    prototypeInstance = createBean(beanName, mbd, args);
                }
                finally {
                    afterPrototypeCreation(beanName);
                }
                bean = getObjectForBeanInstance(prototypeInstance, name, beanName, mbd);
            }
            else {
                // 其他scope的处理
            }
        }
    }
    return (T) bean;
}

createBean() 实际实现调用 AbstractAutowireCapableBeanFactory#doCreateBean 来创建

protected Object doCreateBean(final String beanName, final RootBeanDefinition mbd, final @Nullable Object[] args)
        throws BeanCreationException {
    // Instantiate the bean.
    BeanWrapper instanceWrapper = null;
    // 如果是Singleton bean，则先尝试在cache中获取
    if (mbd.isSingleton()) {
        instanceWrapper = this.factoryBeanInstanceCache.remove(beanName);
    }
    // 创建bean实例
    if (instanceWrapper == null) {
        instanceWrapper = createBeanInstance(beanName, mbd, args);
    }

    final Object bean = instanceWrapper.getWrappedInstance(); // bean实例
    Class<?> beanType = instanceWrapper.getWrappedClass(); // bean实例对应的类型

    Object exposedObject = bean;
    // 【注】填充依赖
    populateBean(beanName, mbd, instanceWrapper);
    // 【注】初始化
    exposedObject = initializeBean(beanName, exposedObject, mbd); // 注意：这里为提前暴露
    // 注册销毁逻辑
    registerDisposableBeanIfNecessary(beanName, bean, mbd);     
    // 暴露实例
    return exposedObject;
}

createBeanInstance() 内调用的是 AbstractAutowireCapableBeanFactory#createBeanInstance

protected BeanWrapper createBeanInstance(String beanName, RootBeanDefinition mbd, @Nullable Object[] args) {
    // 在创建bean前先解析对应的类
    Class<?> beanClass = resolveBeanClass(mbd, beanName);

    // 构建实例的方案 //////////////////////////

    // 情况1：bean从Supplier中产生
    // context.registerBean("myService", MyService.class, () -> new MyService());
    Supplier<?> instanceSupplier = mbd.getInstanceSupplier();
    if (instanceSupplier != null) {
        return obtainFromSupplier(instanceSupplier, beanName);
    }

    // 情况2：bean从工厂方法中产生（譬如@Bean标注在方法上或@Component注解在上）
    // 特别注意：此时因为还没进行属性填充，所以如果在new时手动指定了属性值
    // 而这个属性值又被@Value/@Autowired等Spring注解所标注，则手动指定的属性值会在填充阶段覆盖
    if (mbd.getFactoryMethodName() != null) {
        return instantiateUsingFactoryMethod(beanName, mbd, args);
    }

    // Shortcut when re-creating the same bean...
    boolean resolved = false;
    boolean autowireNecessary = false;
    if (args == null) {
        synchronized (mbd.constructorArgumentLock) {
            if (mbd.resolvedConstructorOrFactoryMethod != null) {
                resolved = true;
                autowireNecessary = mbd.constructorArgumentsResolved;
            }
        }
    }

    // 情况3：构造器创建，譬如@Controller
    if (resolved) {
        // 构造器带有参数，实例化并注入依赖
        if (autowireNecessary) { 
            return autowireConstructor(beanName, mbd, null, null);
        }
        // 无参实例化
        else {
            return instantiateBean(beanName, mbd);
        }
    }

    // ...

    return instantiateBean(beanName, mbd);
}

instantiateUsingFactoryMethod() 内调用的是 ConstructorResolver#instantiate

private Object instantiate(String beanName, RootBeanDefinition mbd,
        @Nullable Object factoryBean, Method factoryMethod, Object[] args) {

    if (System.getSecurityManager() != null) {
        return AccessController.doPrivileged((PrivilegedAction<Object>) () ->
                this.beanFactory.getInstantiationStrategy().instantiate(
                        mbd, beanName, this.beanFactory, factoryBean, factoryMethod, args),
                this.beanFactory.getAccessControlContext());
    }
    else {
        return this.beanFactory.getInstantiationStrategy().instantiate(
                mbd, beanName, this.beanFactory, factoryBean, factoryMethod, args);
    }
}
  • 可以看到 Singleton Bean 实例化逻辑位于 beanFactory.getInstantiationStrategy().instantiate()
  • 目前 Spring 中实例化 Bean 的策略主有两种：
    • SimpleInstantiationStrategy：采用反射进行实例化
    • CglibSubclassingInstantiationStrategy（默认）：通过 CGLIB 生成子类实例化（继承自SimpleInstantiationStrategy），即这时返回的是一个代理

关于InstantiationStrategy：InstantiationStrategy接口

public interface InstantiationStrategy {
    // 无参构造器方式
    Object instantiate(RootBeanDefinition bd, @Nullable String beanName, BeanFactory owner)
            throws BeansException;

    // 有参构造器方式
    Object instantiate(RootBeanDefinition bd, @Nullable String beanName, BeanFactory owner,
            Constructor<?> ctor, Object... args) throws BeansException;

    // 工厂方法方式
    Object instantiate(RootBeanDefinition bd, @Nullable String beanName, BeanFactory owner,
            @Nullable Object factoryBean, Method factoryMethod, Object... args)
            throws BeansException;

}

无参构造器方式 - SimpleInstantiationStrategy#instantiate

@Override
public Object instantiate(RootBeanDefinition bd, @Nullable String beanName, BeanFactory owner) {
    // 不存在方法重写（方法注入）时，采用反射创策略，反之使用CGLIB
    // 特别注意：这里的方法重写不是Java中@Override，而是Spring中的lookup-method和replace-method
    // 1. lookup-method（@Lookup）用于将prototype依赖注入到一个singleton bean对应的getter中。如果不用lookup-method，则注入依赖时spring会将prototype依赖隐式转为singleton
    // 2. replaced-method 用于对bean的方法进行替换
    if (!bd.hasMethodOverrides()) {
        Constructor<?> constructorToUse;
        // ...
        return BeanUtils.instantiateClass(constructorToUse);
    }
    else {
        // Must generate CGLIB subclass.
        return instantiateWithMethodInjection(bd, beanName, owner);
    }
}

有参构造器方式 - SimpleInstantiationStrategy#instantiate

@Override
public Object instantiate(RootBeanDefinition bd, @Nullable String beanName, BeanFactory owner,
                          final Constructor<?> ctor, Object... args) {
    if (!bd.hasMethodOverrides()) {
        // ...
        return BeanUtils.instantiateClass(ctor, args);
    }
    else {
        return instantiateWithMethodInjection(bd, beanName, owner, ctor, args);
    }
}

工厂方法方式 - SimpleInstantiationStrategy#instantiate

@Override
public Object instantiate(RootBeanDefinition bd, @Nullable String beanName, BeanFactory owner,
                          @Nullable Object factoryBean, final Method factoryMethod, Object... args) {
    // ...
    Method priorInvokedFactoryMethod = currentlyInvokedFactoryMethod.get();
    try {
        currentlyInvokedFactoryMethod.set(factoryMethod);
        // 执行工厂方法返回实例
        Object result = factoryMethod.invoke(factoryBean, args);
        if (result == null) {
            result = new NullBean();
        }
        return result;
    }
    finally {
        if (priorInvokedFactoryMethod != null) {
            currentlyInvokedFactoryMethod.set(priorInvokedFactoryMethod);
        }
        else {
            currentlyInvokedFactoryMethod.remove();
        }
    }
}

populateBean() - 填充实例依赖

AbstractAutowireCapableBeanFactory#populateBean

protected void populateBean(String beanName, RootBeanDefinition mbd, @Nullable BeanWrapper bw) {
    // ...
    PropertyValues pvs = (mbd.hasPropertyValues() ? mbd.getPropertyValues() : null);

    // 判断依赖注入的模式：by name（@Qualifier） 或 by type（默认）
    int resolvedAutowireMode = mbd.getResolvedAutowireMode();
    // 如果两个都不是，有两种情况：
    // 1. 没有依赖
    // 2. 默认注入模式（构造器注入）
    if (resolvedAutowireMode == AUTOWIRE_BY_NAME || resolvedAutowireMode == AUTOWIRE_BY_TYPE) {
        MutablePropertyValues newPvs = new MutablePropertyValues(pvs);
        // Add property values based on autowire by name if applicable.
        if (resolvedAutowireMode == AUTOWIRE_BY_NAME) {
            autowireByName(beanName, mbd, bw, newPvs);
        }
        // Add property values based on autowire by type if applicable.
        if (resolvedAutowireMode == AUTOWIRE_BY_TYPE) {
            autowireByType(beanName, mbd, bw, newPvs);
        }
        pvs = newPvs;
    }

    boolean hasInstAwareBpps = hasInstantiationAwareBeanPostProcessors();
    boolean needsDepCheck = (mbd.getDependencyCheck() != AbstractBeanDefinition.DEPENDENCY_CHECK_NONE);

    PropertyDescriptor[] filteredPds = null;
    if (hasInstAwareBpps) {
        if (pvs == null) {
            pvs = mbd.getPropertyValues();
        }
        // 依赖注入逻辑 ///////////////////

        // 通过 InstantiationAwareBeanPostProcessor 实现来执行注入操作
        // 譬如处理 @Autowired、@Value、@Inject 等注解的 AutowiredAnnotationBeanPostProcessor 类
        for (BeanPostProcessor bp : getBeanPostProcessors()) {
            if (bp instanceof InstantiationAwareBeanPostProcessor) {
                InstantiationAwareBeanPostProcessor ibp = (InstantiationAwareBeanPostProcessor) bp;
                PropertyValues pvsToUse = ibp.postProcessProperties(pvs, bw.getWrappedInstance(), beanName);
                if (pvsToUse == null) {
                    if (filteredPds == null) {
                        filteredPds = filterPropertyDescriptorsForDependencyCheck(bw, mbd.allowCaching);
                    }
                    pvsToUse = ibp.postProcessPropertyValues(pvs, filteredPds, bw.getWrappedInstance(), beanName);
                    if (pvsToUse == null) {
                        return;
                    }
                }
                pvs = pvsToUse;
            }
        }
    }
}

initializeBean() - 生命周期模板方法回调

AbstractAutowireCapableBeanFactory#initializeBean

protected Object initializeBean(final String beanName, final Object bean, @Nullable RootBeanDefinition mbd) {
    if (System.getSecurityManager() != null) {
        AccessController.doPrivileged((PrivilegedAction<Object>) () -> {
            invokeAwareMethods(beanName, bean);
            return null;
        }, getAccessControlContext());
    }
    else {
        // 步骤1：回调特定的Aware接口方法
         // BeanNameAware -> BeanClassLoaderAware -> BeanFactoryAware
        invokeAwareMethods(beanName, bean);
    }

    Object wrappedBean = bean;
    if (mbd == null || !mbd.isSynthetic()) {
        // 步骤2：以下方法位于 AbstractAutowireCapableBeanFactory ，有以下功能
        // 1）回调当前bean实现的ApplicationContextAware接口方法
        // 2）回调当前bean实现的ServletContextAware接口方法
        // 3) 回调当前bean的依赖所实现的BeanPostProcessor方法
        // 4）回调CommonAnnotationBeanPostProcessor处理@PostConstruct
        wrappedBean = applyBeanPostProcessorsBeforeInitialization(wrappedBean, beanName);

        // BeanPostProcessor的作用
        // 1. postProcessAfterInitialization()：在当前bean的使用bean初始化前调用（@PostConstruct前）。注意，此时当前bean已经被注入到使用bean中了
        // 2. postProcessAfterInitialization()：在当前bean的使用bean初始化后调用
        // 3. BeanPostProcessor回调时可以获取到使用bean的实例引用
    }

    try {
        // 步骤3：回调初始化方法
        // 1）先回调通过InitializingBean的afterPropertiesSet实现的初始化方法
        // 2) 再回调init-method指定的初始化方法
        invokeInitMethods(beanName, wrappedBean, mbd);
    }
    catch (Throwable ex) {
        throw new BeanCreationException(
                (mbd != null ? mbd.getResourceDescription() : null),
                beanName, "Invocation of init method failed", ex);
    }
    if (mbd == null || !mbd.isSynthetic()) {
        // 步骤4：回调BeanPostProcessor的后置方法（postProcessAfterInitialization）
        wrappedBean = applyBeanPostProcessorsAfterInitialization(wrappedBean, beanName);
    }

    return wrappedBean;
}

AbstractAutowireCapableBeanFactory#invokeInitMethods

protected void invokeInitMethods(String beanName, final Object bean, @Nullable RootBeanDefinition mbd)
    throws Throwable {

    boolean isInitializingBean = (bean instanceof InitializingBean);
    // 先回调通过InitializingBean的afterPropertiesSet实现的初始化方法，如果有的话
    // 注意，如果afterPropertiesSet方法被标注了@PostConstruct注解，则会提前执行而不是在这里执行
    if (isInitializingBean && (mbd == null || !mbd.isExternallyManagedInitMethod("afterPropertiesSet"))) {
        // ...
        ((InitializingBean) bean).afterPropertiesSet();
    }

    // 再回调init-method指定的初始化方法，如果有的话
    if (mbd != null && bean.getClass() != NullBean.class) {
        String initMethodName = mbd.getInitMethodName();
        if (StringUtils.hasLength(initMethodName) &&
            !(isInitializingBean && "afterPropertiesSet".equals(initMethodName)) &&
            !mbd.isExternallyManagedInitMethod(initMethodName)) {
            invokeCustomInitMethod(beanName, bean, mbd);
        }
    }
}

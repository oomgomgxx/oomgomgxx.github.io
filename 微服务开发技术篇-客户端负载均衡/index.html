什么是客户端负载均衡

看到负载均衡很多人首先想到的是Nginx。Nginx是一个代理服务器，提供正反两种模式的代理服务。其中负载均衡就是基于反向代理来实现的，这种负载均衡方式可以理解为后端负载均衡，即客户端请求到来后Nginx以反向代理的身份来处理客户端请求

从上图不难看出看出负载均衡算法是在反向代理服务器上实现的。而本文要谈的是客户端负载均衡，它是在客户端上实现的。客户端负载均衡和后端负载均衡是互不冲突的

public class Test {
    private static Random random = null;
    private static List<String> userServices = null;

        random = new Random();

        // 设置权重（权重大的话被调用的几率就大）
        weightMap = new TreeMap<>();
        weightMap.put("192.168.31.100", 2);

        // 用于存放加权后的服务列表
        userServices = new ArrayList<>();
        Iterator<Map.Entry<String, Integer>> iterator = weightMap.entrySet().iterator();
        while (iterator.hasNext()) {
            Map.Entry<String, Integer> next = iterator.next();
            Integer count = next.getValue();
            if(count <= 0) count = 1;
            for (int i = 0; i < count; i++) { // 加权操作
                userServices.add(next.getKey());
            }
        }
    }

算法实现

public class Test implements Runnable {
    private List<Connect> connects = null;
    public Test(List<Connect> connects) {
        this.connects = connects;
    }

    @Override
    public void run() {
        // 获取连接
        Connect connect = getConnect();
        try {
            // 模拟处理业务时间
            TimeUnit.SECONDS.sleep(new Random().nextInt(10));
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
        // 释放连接
        releaseConnect(connect);
    }

    // 释放连接
    public void releaseConnect(Connect conn) {
        conn.getCount().decrementAndGet();
    }

    // 获取连接
    public synchronized Connect getConnect() {
        // 获取连接前先按照连接次数排序（冒泡算法），找出最少连接的服务
        for (int i = 0; i < connects.size() - 1; i++) {
            for (int j = 0; j < connects.size()-i-1; j++) {
                Connect connect = connects.get(j);
                Connect connect2 = connects.get(j+1);
                if (connect.getCount().get() > connect2.getCount().get()) {
                    connects.set(j, connect2);
                    connects.set(j + 1, connect);
                }
            }
        }
        Connect conn = connects.get(0);
        conn.getCount().incrementAndGet();
        return conn;
    }

    public static void main(String[] args) {
        // 服务列表
        List<Connect> list = new ArrayList<>();
        list.add(new Connect("user-service", "192.168.31.101"));
        list.add(new Connect("user-service", "192.168.31.102"));
        list.add(new Connect("user-service", "192.168.31.103"));

        // 50个请求
        ExecutorService service = Executors.newFixedThreadPool(20);
        T5 task = new T5(list);
        for (int i = 0; i <50 ; i++) {
            service.submit(task);
        }

    }
}

一致性哈希

public class Test {

    /**
     * 简易版一致性哈希
     * @param client 客户单key
     * @param serviceNode 服务列表
     * @return
     */
    public static String getService(String client, String[] serviceNode) {

        // 预先将服务节点固定在哈希环上（这里使用TreeMap来实现哈希环）
        TreeMap<Integer, String> treeMap = new TreeMap<>();
        for (int i = 0; i < serviceNode.length; i++) {
            String service = serviceNode[i];
            // 加大间隔，模拟让节点更加离散
            treeMap.put(service.hashCode() * (i+1), service);
        }

        // 获取等于或大于 client.hashCode() 的服务节点
        SortedMap<Integer, String> sm = treeMap.tailMap(client.hashCode());
        try {
            return treeMap.get(sm.firstKey());
        } catch (NoSuchElementException e) {
            return treeMap.get(treeMap.firstKey());
        }
    }

    public static void main(String[] args) {

        String[] clients = new String[]{"33.33.33.33", "66.66.66.66", "123.123.123.123"};
        String[] userServices = new String[]{"192.168.31.100",
                "192.168.31.102",

BaseLoadBalancer 源码

BaseLoadBalancer 是 Ribbon 负载均衡器基础实现

public class BaseLoadBalancer extends AbstractLoadBalancer {
    // 默认负载均衡策略RoundRobinRule
    private final static IRule DEFAULT_RULE = new RoundRobinRule();
    protected IRule rule = DEFAULT_RULE;

    // 所有服务
    @Monitor(name = PREFIX + "AllServerList", type = DataSourceType.INFORMATIONAL)
    protected volatile List<Server> allServerList = Collections
            .synchronizedList(new ArrayList<Server>());
    // 健康服务
    @Monitor(name = PREFIX + "UpServerList", type = DataSourceType.INFORMATIONAL)
    protected volatile List<Server> upServerList = Collections
            .synchronizedList(new ArrayList<Server>());

    private static final String DEFAULT_NAME = "default";
    public BaseLoadBalancer() {
        this.name = DEFAULT_NAME;
        this.ping = null;
        setRule(DEFAULT_RULE);
        setupPingTask();
        lbStats = new LoadBalancerStats(DEFAULT_NAME);
    }
    // 设置用来检测服务是否可用的Ping任务
    void setupPingTask() {
        lbTimer.schedule(new PingTask(), 0, pingIntervalSeconds * 1000);
        forceQuickPing();
    }
    // Ping任务
    class PingTask extends TimerTask {
        public void run() {
            new Pinger(pingStrategy).runPinger();
        }
    }
    // 获取负载均衡后的服务地址
    public String choose(Object key) {
        Server svr = rule.choose(key);
        return ((svr == null) ? null : svr.getId());
    }
    // 获取负载均衡后的服务
    public Server chooseServer(Object key) {
       if (counter == null) {
           counter = createCounter();
       }
       counter.increment();
       return rule.choose(key);
    }
    // 添加服务到服务列表
    public void addServer(Server newServer) {
        ArrayList<Server> newList = new ArrayList<Server>();
        newList.addAll(allServerList);
        newList.add(newServer);
        // 添加到 allServerList 和 upServerList中
        setServersList(newList);
    }
}

public interface ServerListUpdater {
    public interface UpdateAction {
        void doUpdate();
    }
    // 启动更新器
    void start(UpdateAction updateAction);
    // 停止更新器
    void stop();
    // 获取核心线程数

public class DynamicServerListLoadBalancer<T extends Server> extends BaseLoadBalancer {
    // 服务列表
    volatile ServerList<T> serverListImpl;
    // 服务列表过滤器
    volatile ServerListFilter<T> filter;
    // 更新服务列表
    protected final ServerListUpdater.UpdateAction updateAction 
        = new ServerListUpdater.UpdateAction() {
        @Override
        public void doUpdate() {
            updateListOfServers();
        }
    };    
    // 服务列表更新器
    protected volatile ServerListUpdater serverListUpdater;
    // 启动服务列表更新器
    public void enableAndInitLearnNewServersFeature() {
        serverListUpdater.start(updateAction);
    }
    // 停止服务列表更新器
    public void stopServerListRefreshing() {
        if (serverListUpdater != null) {
            serverListUpdater.stop();
        }
    }
    // 获取服务列表更新器核心线程数
    @Monitor(name="NumThreads", type=DataSourceType.GAUGE)
    public int getCoreThreads() {
        return serverListUpdater.getCoreThreads();
    }
    // 更新服务列表
    @VisibleForTesting
    public void updateListOfServers() {
        List<T> servers = new ArrayList<T>();
        // 获取需要更新的服务列表
        servers = serverListImpl.getUpdatedListOfServers();
        // 进行过滤
        servers = filter.getFilteredListOfServers(servers);
        updateAllServerList(servers);
    }
    protected void updateAllServerList(List<T> ls) {
        if (serverListUpdateInProgress.compareAndSet(false, true)) {
            try {
                // 更新成功设置为可用
                for (T s : ls) {
                    s.setAlive(true);
                }
                setServersList(ls);
                super.forceQuickPing();
            } finally {
                serverListUpdateInProgress.set(false);
            }
        }
    }
}

ZoneAwareLoadBalancer 源码

  • 继承自 DynamicServerListLoadBalancer
  • 引入了 Zone 来划分服务
  • 重写了 setRule 方法 （DynamicServerListLoadBalancer 默认使用 BaseLoadBalancer 中的 RoundRobinRule）
public class ZoneAwareLoadBalancer<T extends Server> extends DynamicServerListLoadBalancer<T> {
    // String：Zone
    // BaseLoadBalancer:对应的负载均衡器
    private ConcurrentHashMap<String, BaseLoadBalancer> balancers = new ConcurrentHashMap<String, BaseLoadBalancer>();

    // 设置负载均衡策略
    @Override
    public void setRule(IRule rule) {
        super.setRule(rule);
        if (balancers != null) {
            for (String zone: balancers.keySet()) {
                // 设置到所有zone中
                balancers.get(zone).setRule(cloneRule(rule));
            }
        }
    }

    // 获取某个zone的负载均衡器
    @VisibleForTesting
    BaseLoadBalancer getLoadBalancer(String zone) {
        zone = zone.toLowerCase();
        BaseLoadBalancer loadBalancer = balancers.get(zone);
        if (loadBalancer == null) {
            IRule rule = cloneRule(this.getRule());
            loadBalancer = new BaseLoadBalancer(this.getName() + "_" + zone, rule, this.getLoadBalancerStats());
            BaseLoadBalancer prev = balancers.putIfAbsent(zone, loadBalancer);
            if (prev != null) {
                loadBalancer = prev;
            }
        } 
        return loadBalancer;        
    }

    // 获取负载均衡后的服务
    @Override
    public Server chooseServer(Object key) {
        Server server = null;
        try {
            LoadBalancerStats lbStats = getLoadBalancerStats();
            Map<String, ZoneSnapshot> zoneSnapshot = ZoneAvoidanceRule.createSnapshot(lbStats);
            // 获取可用的zone
            Set<String> availableZones = ZoneAvoidanceRule.getAvailableZones(zoneSnapshot, triggeringLoad.get(), triggeringBlackoutPercentage.get());
            if (availableZones != null &&  availableZones.size() < zoneSnapshot.keySet().size()) {
                // 在可用zone中随机选择一个
                String zone = ZoneAvoidanceRule.randomChooseZone(zoneSnapshot, availableZones);
                if (zone != null) {
                    // 获取该zone的负载均衡器
                    BaseLoadBalancer zoneLoadBalancer = getLoadBalancer(zone);
                    // 返回经过负载均衡的服务
                    server = zoneLoadBalancer.chooseServer(key);
                }
            }
        } 
        return server;
    }
}

    return ThreadLocalRandom.current().nextInt(serverCount);

// 返回负载均衡后的服务
@Override
public Server choose(Object key) {
    return choose(getLoadBalancer(), key);

// 获取服务过程
public Server choose(ILoadBalancer lb, Object key) {
    Server server = null;
    while (server == null) { // 死循环获取直到不为null
        if (Thread.interrupted()) {
            return null;
        }
        List<Server> upList = lb.getReachableServers(); // 可用服务列表
        List<Server> allList = lb.getAllServers(); // 所有服务
        int serverCount = allList.size();
        if (serverCount == 0) {
            return null;
        }
        // 获取随机下标
        int index = chooseRandomInt(serverCount);
        server = upList.get(index); // 从可用服务列表中获取
        if (server.isAlive()) {
            return (server);
        }
    }
    return server;
}

RoundRibbonRule：轮询策略

// 获取服务
public Server choose(ILoadBalancer lb, Object key) {
    Server server = null;
    int count = 0;
    while (server == null && count++ < 10) {
        List<Server> reachableServers = lb.getReachableServers();
        List<Server> allServers = lb.getAllServers();
        int upCount = reachableServers.size(); // up状态的服务数量
        int serverCount = allServers.size(); // 服务列表中的服务数量
        // 轮询获取服务下标
        int nextServerIndex = incrementAndGetModulo(serverCount);
        server = allServers.get(nextServerIndex);
        // 得到结果后如果是可用的就返回
        if (server.isAlive() && (server.isReadyToServe())) {
            return (server);
        }
        // Next.
        server = null;
    }
    return server;
}

RetryRule：基于轮询的重试策略

IRule subRule = new RoundRobinRule();
// 获取服务
public Server choose(ILoadBalancer lb, Object key) {
    long requestTime = System.currentTimeMillis();
    long deadline = requestTime + maxRetryMillis;
    Server answer = null;
    // 先用轮询获取
    answer = subRule.choose(key);
    // 当获取到的服务不可用则进入重试
    if (((answer == null) || (!answer.isAlive()))
        && (System.currentTimeMillis() < deadline)) {
        InterruptTask task = new InterruptTask(deadline
                                    - System.currentTimeMillis());
        while (!Thread.interrupted()) {
            answer = subRule.choose(key);
            if (((answer == null) || (!answer.isAlive()))
                && (System.currentTimeMillis() < deadline)) {
                Thread.yield();
            } else {
                break;
            }
        }
        task.cancel();
    }
    return answer;
}

// 过滤轮询
public Optional<Server> chooseRoundRobinAfterFiltering(List<Server> servers) {
    List<Server> eligible = getEligibleServers(servers);
    if (eligible.size() == 0) {
        return Optional.absent();
    }
    return Optional.of(eligible.get(incrementAndGetModulo(eligible.size())));
}

// 筛选zone
public static Set<String> getAvailableZones(
    Map<String, ZoneSnapshot> snapshot, double triggeringLoad,
    double triggeringBlackoutPercentage) {
    // 获取可用zone
    Set<String> availableZones = new HashSet<String>(snapshot.keySet());
    if (availableZones.size() == 1) {
        return availableZones; // 如果只有一个直接返回zone
    }
    Set<String> worstZones = new HashSet<String>();
    double maxLoadPerServer = 0;
    boolean limitedZoneAvailability = false;
    for (Map.Entry<String, ZoneSnapshot> zoneEntry : snapshot.entrySet()) {
        String zone = zoneEntry.getKey();
        ZoneSnapshot zoneSnapshot = zoneEntry.getValue();
        int instanceCount = zoneSnapshot.getInstanceCount();
        // 如果zone没有服务则移除zone
        if (instanceCount == 0) {
            availableZones.remove(zone);
            limitedZoneAvailability = true;
        } else { // zone中有服务
            double loadPerServer = zoneSnapshot.getLoadPerServer();
            if (((double) zoneSnapshot.getCircuitTrippedCount())
                / instanceCount >= triggeringBlackoutPercentage
                || loadPerServer < 0) {
                availableZones.remove(zone);
                limitedZoneAvailability = true;
            } else {
                if (Math.abs(loadPerServer - maxLoadPerServer) < 0.000001d) {
                    worstZones.add(zone);
                } else if (loadPerServer > maxLoadPerServer) {
                    maxLoadPerServer = loadPerServer;
                    worstZones.clear();
                    worstZones.add(zone);
                }
            }
        }
    }
   if (maxLoadPerServer < triggeringLoad && !limitedZoneAvailability) {
            // zone override is not needed here
            return availableZones;
        }
    String zoneToAvoid = randomChooseZone(snapshot, worstZones);
    if (zoneToAvoid != null) {
        availableZones.remove(zoneToAvoid);
    }
    return availableZones;
}
// 获取服务
@Override
public Server choose(Object key) {
    ILoadBalancer lb = getLoadBalancer();
    // 过滤之后轮询获取服务
    Optional<Server> server = getPredicate().chooseRoundRobinAfterFiltering(lb.getAllServers(), key);
    if (server.isPresent()) {
        return server.get();
    } else {
        return null;
    }       
}

@Configuration

        LoadBalancerClient loadBalancerClient,
        LoadBalancerRequestFactory requestFactory) {
        return new LoadBalancerInterceptor(loadBalancerClient, requestFactory);
    }

    // 将负载均衡拦截器添加到RestTemplate中
    @Bean
    @ConditionalOnMissingBean
    public RestTemplateCustomizer restTemplateCustomizer(
        // 注入负载均衡拦截器
        final LoadBalancerInterceptor loadBalancerInterceptor) {
        return restTemplate -> {
            // 获取RestTemplate中的拦截器器列表
            List<ClientHttpRequestInterceptor> list = new ArrayList<>(
                restTemplate.getInterceptors());
            // 将负载均衡拦截器添加到列表中
            list.add(loadBalancerInterceptor);
            restTemplate.setInterceptors(list);
        };
    }

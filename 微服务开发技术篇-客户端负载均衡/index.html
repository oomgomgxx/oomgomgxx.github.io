public class T5 implements Runnable {
    private List<Connect> connects = null;
    public T5(List<Connect> connects) {
        this.connects = connects;
    }

    @Override
    public void run() {
        // иҺ·еҸ–иҝһжҺҘ
        Connect connect = getConnect();
        try {
            // жЁЎжӢҹеӨ„зҗҶдёҡеҠЎж—¶й—ҙ
            TimeUnit.SECONDS.sleep(new Random().nextInt(10));
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
        // йҮҠж”ҫиҝһжҺҘ
        releaseConnect(connect);
    }

    // йҮҠж”ҫиҝһжҺҘ
    public void releaseConnect(Connect conn) {
        conn.getCount().decrementAndGet();
    }

    // иҺ·еҸ–иҝһжҺҘ
    public synchronized Connect getConnect() {
        // иҺ·еҸ–еүҚе…ҲжҢүз…§и°ғз”Ёж¬Ўж•°жҺ’еәҸ - еҶ’жіЎз®—жі•
        for (int i = 0; i < connects.size() - 1; i++) {
            for (int j = 0; j < connects.size()-i-1; j++) {
                Connect connect = connects.get(j);
                Connect connect2 = connects.get(j+1);
                if (connect.getCount().get() > connect2.getCount().get()) {
                    connects.set(j, connect2);
                    connects.set(j + 1, connect);
                }
            }
        }
        Connect conn = connects.get(0);
        conn.getCount().incrementAndGet();
        System.out.println("иҙҹиҪҪеқҮиЎЎи°ғз”Ё:" + conn.getIp() + ",еҪ“еүҚи°ғз”Ёж•°:" + conn.getCount().get());
        return conn;
    }

    public static void main(String[] args) {
        List<Connect> list = new ArrayList<>();
        list.add(new Connect("user-service", "192.168.31.101"));
        list.add(new Connect("user-service", "192.168.31.102"));
        list.add(new Connect("user-service", "192.168.31.103"));

        ExecutorService service = Executors.newFixedThreadPool(20);
        T5 task = new T5(list);
        // 50дёӘиҝһжҺҘ
        for (int i = 0; i <50 ; i++) {
            service.submit(task);
        }

    }
}

public class T7 {

    /**
     * з®Җжҳ“зүҲдёҖиҮҙжҖ§е“ҲеёҢ
     * @param client е®ўжҲ·еҚ•key
     * @param serviceNode жңҚеҠЎеҲ—иЎЁ
     * @return
     */
    public static String getService(String client, String[] serviceNode) {

        // йў„е…Ҳе°ҶиҠӮзӮ№ж”ҫеңЁе“ҲеёҢзҺҜдёҠпјҲиҝҷйҮҢзҡ„зҺҜдҪҝз”ЁTreeMapжқҘе®һзҺ°пјү
        TreeMap<Integer, String> treeMap = new TreeMap<>();
        for (int i = 0; i < serviceNode.length; i++) {
            String service = serviceNode[i];
            // еҠ еӨ§й—ҙйҡ”пјҢжЁЎжӢҹи®©иҠӮзӮ№жӣҙеҠ зҰ»ж•Ј
            treeMap.put(service.hashCode() * (i+1), service);
        }

        // иҺ·еҸ–зӯүдәҺеӨ§дәҺclient.hashCode()зҡ„иҠӮзӮ№
        SortedMap<Integer, String> sm = treeMap.tailMap(client.hashCode());
        try {
            return treeMap.get(sm.firstKey());
        } catch (NoSuchElementException e) {
            return treeMap.get(treeMap.firstKey());
        }
    }

    public static void main(String[] args) {

        String[] clients = new String[]{"33.33.33.33", "66.66.66.66", "123.123.123.123"};
        String[] userServices = new String[]{"192.168.31.100",
                "192.168.31.102",
                "192.168.31.103",
                "12.4.55.13",
                "192.168.31.102",
                "16.6.3.2",
                "123.454.33.77",
                "2.2.2.2"};

        // жЁЎжӢҹжҜҸдёӘз”ЁжҲ·еҲҶеҲ«и°ғз”Ё3ж¬Ў
        for (int i = 0; i < 3; i++) {
            for (String client : clients) {
                String service = getService(client, userServices);
                System.out.println(service);
            }
        }

        /**
         * и°ғз”Ёз»“жһңпјҡ
             192.168.31.103
             192.168.31.102
             16.6.3.2
             192.168.31.103
             192.168.31.102
             16.6.3.2
             192.168.31.103
             192.168.31.102
             16.6.3.2
         */
    }
}

public class ZoneAwareLoadBalancer<T extends Server> extends DynamicServerListLoadBalancer<T> {
    // StringпјҡZone
    // BaseLoadBalancer:еҜ№еә”зҡ„иҙҹиҪҪеқҮиЎЎеҷЁ
    private ConcurrentHashMap<String, BaseLoadBalancer> balancers = new ConcurrentHashMap<String, BaseLoadBalancer>();

    // и®ҫзҪ®иҙҹиҪҪеқҮиЎЎзӯ–з•Ҙ
    @Override
    public void setRule(IRule rule) {
        super.setRule(rule);
        if (balancers != null) {
            for (String zone: balancers.keySet()) {
                // и®ҫзҪ®еҲ°жүҖжңүzoneдёӯ
                balancers.get(zone).setRule(cloneRule(rule));
            }
        }
    }

    // иҺ·еҸ–жҹҗдёӘzoneзҡ„иҙҹиҪҪеқҮиЎЎеҷЁ
    @VisibleForTesting
    BaseLoadBalancer getLoadBalancer(String zone) {
        zone = zone.toLowerCase();
        BaseLoadBalancer loadBalancer = balancers.get(zone);
        if (loadBalancer == null) {
            IRule rule = cloneRule(this.getRule());
            loadBalancer = new BaseLoadBalancer(this.getName() + "_" + zone, rule, this.getLoadBalancerStats());
            BaseLoadBalancer prev = balancers.putIfAbsent(zone, loadBalancer);
            if (prev != null) {
                loadBalancer = prev;
            }
        } 
        return loadBalancer;        
    }
    // иҺ·еҸ–Server
    @Override
    public Server chooseServer(Object key) {
        Server server = null;
        try {
            LoadBalancerStats lbStats = getLoadBalancerStats();
            Map<String, ZoneSnapshot> zoneSnapshot = ZoneAvoidanceRule.createSnapshot(lbStats);
            // иҺ·еҸ–еҸҜз”Ёзҡ„zone
            Set<String> availableZones = ZoneAvoidanceRule.getAvailableZones(zoneSnapshot, triggeringLoad.get(), triggeringBlackoutPercentage.get());
            if (availableZones != null &&  availableZones.size() < zoneSnapshot.keySet().size()) {
                // йҡҸжңәз®—зқҖдёҖдёӘZone
                String zone = ZoneAvoidanceRule.randomChooseZone(zoneSnapshot, availableZones);
                if (zone != null) {
                    // иҺ·еҸ–иҜҘzoneзҡ„иҙҹиҪҪеқҮиЎЎеҷЁ
                    BaseLoadBalancer zoneLoadBalancer = getLoadBalancer(zone);
                    // иҝ”еӣһз»ҸиҝҮиҙҹиҪҪеқҮиЎЎзҡ„жңҚеҠЎ
                    server = zoneLoadBalancer.chooseServer(key);
                }
            }
        } catch (Exception e) {
            logger.error("Error choosing server using zone aware logic for load balancer={}", name, e);
        }
        if (server != null) {
            return server;
        } else {
            logger.debug("Zone avoidance logic is not invoked.");
            return super.chooseServer(key);
        }
    }
}

public Server choose(ILoadBalancer lb, Object key) {

    while (server == null) {
        if (Thread.interrupted()) {
            return null;
        }
        List<Server> upList = lb.getReachableServers();
        List<Server> allList = lb.getAllServers();
        int serverCount = allList.size();
        if (serverCount == 0) {
            return null;
        }
        // иҺ·еҸ–йҡҸжңәдёӢж Ү
        int index = chooseRandomInt(serverCount);
        server = upList.get(index);
        if (server.isAlive()) {
            return (server);
        }
    }
    return server;
}

                                    - System.currentTimeMillis());

            answer = subRule.choose(key);
            if (((answer == null) || (!answer.isAlive()))

            } else {
                break;
            }
        }
        task.cancel();
    }
    return answer;
}

public static Set<String> getAvailableZones(
    Map<String, ZoneSnapshot> snapshot, double triggeringLoad,
    double triggeringBlackoutPercentage) {
    // иҺ·еҸ–еҸҜз”Ёzone
    Set<String> availableZones = new HashSet<String>(snapshot.keySet());
    if (availableZones.size() == 1) {
        return availableZones; // еҰӮжһңеҸӘжңүдёҖдёӘзӣҙжҺҘиҝ”еӣһzone
    }
    Set<String> worstZones = new HashSet<String>();
    double maxLoadPerServer = 0;
    boolean limitedZoneAvailability = false;
    for (Map.Entry<String, ZoneSnapshot> zoneEntry : snapshot.entrySet()) {
        String zone = zoneEntry.getKey();
        ZoneSnapshot zoneSnapshot = zoneEntry.getValue();
        int instanceCount = zoneSnapshot.getInstanceCount();
        // еҰӮжһңzoneжІЎжңүжңҚеҠЎеҲҷз§»йҷӨzone
        if (instanceCount == 0) {
            availableZones.remove(zone);
            limitedZoneAvailability = true;
        } else { // zoneдёӯжңүжңҚеҠЎ
            double loadPerServer = zoneSnapshot.getLoadPerServer();
            if (((double) zoneSnapshot.getCircuitTrippedCount())
                / instanceCount >= triggeringBlackoutPercentage
                || loadPerServer < 0) {
                availableZones.remove(zone);
                limitedZoneAvailability = true;
            } else {
                if (Math.abs(loadPerServer - maxLoadPerServer) < 0.000001d) {
                    worstZones.add(zone);
                } else if (loadPerServer > maxLoadPerServer) {
                    maxLoadPerServer = loadPerServer;
                    worstZones.clear();
                    worstZones.add(zone);
                }
            }
        }
    }
   if (maxLoadPerServer < triggeringLoad && !limitedZoneAvailability) {
            // zone override is not needed here
            return availableZones;
        }
    String zoneToAvoid = randomChooseZone(snapshot, worstZones);
    if (zoneToAvoid != null) {
        availableZones.remove(zoneToAvoid);
    }
    return availableZones;
}

    client.setLoadBalancer(loadBalancer); // иҙҹиҪҪеқҮиЎЎзұ»пјҢеҢ…еҗ«иҙҹиҪҪеқҮиЎЎзӣёе…іж“ҚдҪң

}

public Server choose(ILoadBalancer lb, Object key) {
    Server server = null;
    int count = 0;
    while (server == null && count++ < 10) {
        List<Server> reachableServers = lb.getReachableServers();
        List<Server> allServers = lb.getAllServers();
        int upCount = reachableServers.size(); // upзҠ¶жҖҒзҡ„жңҚеҠЎж•°йҮҸ
        int serverCount = allServers.size(); // жңҚеҠЎеҲ—иЎЁдёӯзҡ„жңҚеҠЎж•°йҮҸ
        // иҪ®иҜў
        int nextServerIndex = incrementAndGetModulo(serverCount);
        server = allServers.get(nextServerIndex);
        // еҫ—еҲ°з»“жһңеҗҺеҰӮжһңжҳҜеҸҜз”Ёзҡ„е°ұиҝ”еӣһ
        if (server.isAlive() && (server.isReadyToServe())) {
            return (server);
        }
        // Next.
        server = null;
    }
    return server;
}
